# 高级模板技术详细解析

> **核心结论**：CRTP实现静态多态零开销，模板元编程在编译期完成计算，Policy-Based Design提供灵活组合，C++20 Concepts革新了模板约束方式。这些高级技术是构建高性能、可扩展C++库的基石。

---

## 1. Why — 为什么需要高级模板技术

**结论先行**：高级模板技术让C++在编译期完成更多工作，实现零开销抽象、静态多态和高度可定制的库设计。

### 1.1 高级模板技术的价值

```cpp
// 传统运行时多态（有虚函数开销）
class Interface {
public:
    virtual void algorithm() = 0;
    virtual ~Interface() = default;
};

// CRTP静态多态（零开销）
template<typename Derived>
class Interface {
public:
    void algorithm() {
        static_cast<Derived*>(this)->implementation();
    }
};
```

### 1.2 应用场景对比

| 技术 | 应用场景 | 优势 |
|-----|---------|------|
| **CRTP** | 静态多态、代码复用 | 零运行时开销 |
| **模板元编程** | 编译期计算、类型操作 | 运行时零开销 |
| **Policy-Based Design** | 可定制行为 | 灵活组合，无继承开销 |
| **Concepts** | 模板约束 | 清晰错误信息，自文档化 |

## 2. What — 高级模板技术详解

### 2.1 CRTP（Curiously Recurring Template Pattern）

```cpp
// C++11/14/17/20/23 通用
#include <iostream>
#include <memory>

// ============== CRTP基本形式 ==============
template<typename Derived>
class Base {
public:
    void interface() {
        // 静态转换为派生类，调用派生类实现
        static_cast<Derived*>(this)->implementation();
    }
    
    void default_implementation() {
        std::cout << "Base默认实现" << std::endl;
    }
};

class Derived1 : public Base<Derived1> {
public:
    void implementation() {
        std::cout << "Derived1实现" << std::endl;
    }
};

class Derived2 : public Base<Derived2> {
public:
    void implementation() {
        std::cout << "Derived2实现" << std::endl;
    }
};

// ============== CRTP实现计数器 ==============
template<typename Derived>
class Counter {
public:
    Counter() { ++count_; }
    Counter(const Counter&) { ++count_; }
    ~Counter() { --count_; }
    
    static std::size_t count() { return count_; }
    
private:
    static std::size_t count_;
};

template<typename Derived>
std::size_t Counter<Derived>::count_ = 0;

class Widget : public Counter<Widget> {};
class Gadget : public Counter<Gadget> {};

// ============== CRTP实现Mixin ==============
// 可比较Mixin
template<typename Derived>
class Comparable {
public:
    bool operator!=(const Derived& other) const {
        return !static_cast<const Derived*>(this)->operator==(other);
    }
    
    bool operator<(const Derived& other) const {
        return static_cast<const Derived*>(this)->compare(other) < 0;
    }
    
    bool operator>(const Derived& other) const {
        return other < *static_cast<const Derived*>(this);
    }
    
    bool operator<=(const Derived& other) const {
        return !(static_cast<const Derived*>(this)->operator>(other));
    }
    
    bool operator>=(const Derived& other) const {
        return !(static_cast<const Derived*>(this)->operator<(other));
    }
};

class Value : public Comparable<Value> {
public:
    explicit Value(int v) : value_(v) {}
    
    bool operator==(const Value& other) const {
        return value_ == other.value_;
    }
    
    int compare(const Value& other) const {
        return value_ - other.value_;
    }
    
private:
    int value_;
};

// ============== CRTP实现enable_shared_from_this ==============
template<typename Derived>
class EnableSharedFromThis {
public:
    std::shared_ptr<Derived> shared_from_this() {
        return std::static_pointer_cast<Derived>(weak_this_.lock());
    }
    
    std::shared_ptr<const Derived> shared_from_this() const {
        return std::static_pointer_cast<const Derived>(weak_this_.lock());
    }
    
private:
    template<typename T>
    friend class std::enable_shared_from_this;
    
    mutable std::weak_ptr<Derived> weak_this_;
};

// 简化版实现
template<typename Derived>
class MyEnableSharedFromThis {
public:
    std::shared_ptr<Derived> shared_from_this() {
        return static_cast<Derived*>(this)->shared_from_this_impl();
    }
    
protected:
    std::weak_ptr<Derived> weak_this_;
};

class SharedObject : public MyEnableSharedFromThis<SharedObject> {
public:
    std::shared_ptr<SharedObject> get_shared() {
        return shared_from_this();
    }
    
    std::shared_ptr<SharedObject> shared_from_this_impl() {
        return weak_this_.lock();
    }
    
    template<typename T>
    friend std::shared_ptr<T> make_shared_object();
};

template<typename T>
std::shared_ptr<T> make_shared_object() {
    auto ptr = std::make_shared<T>();
    ptr->weak_this_ = ptr;
    return ptr;
}

// ============== CRTP实现对象克隆 ==============
template<typename Derived>
class Cloneable {
public:
    std::unique_ptr<Derived> clone() const {
        return std::unique_ptr<Derived>(
            static_cast<const Derived*>(this)->clone_impl()
        );
    }
};

class Document : public Cloneable<Document> {
public:
    Document(const std::string& content) : content_(content) {}
    
    Document* clone_impl() const {
        return new Document(*this);
    }
    
    void print() const {
        std::cout << "Document: " << content_ << std::endl;
    }
    
private:
    std::string content_;
};

// ============== 测试 ==============
int main() {
    std::cout << "=== CRTP静态多态 ===" << std::endl;
    Derived1 d1;
    Derived2 d2;
    d1.interface();
    d2.interface();
    
    std::cout << "\n=== CRTP计数器 ===" << std::endl;
    {
        Widget w1, w2;
        Gadget g1;
        std::cout << "Widget count: " << Widget::count() << std::endl;
        std::cout << "Gadget count: " << Gadget::count() << std::endl;
    }
    std::cout << "Widget count: " << Widget::count() << std::endl;
    
    std::cout << "\n=== CRTP Mixin ===" << std::endl;
    Value v1(10), v2(20);
    std::cout << "v1 == v2: " << (v1 == v2) << std::endl;
    std::cout << "v1 != v2: " << (v1 != v2) << std::endl;
    std::cout << "v1 < v2: " << (v1 < v2) << std::endl;
    std::cout << "v1 > v2: " << (v1 > v2) << std::endl;
    
    std::cout << "\n=== CRTP克隆 ===" << std::endl;
    Document doc("Hello, CRTP!");
    auto cloned = doc.clone();
    cloned->print();
    
    return 0;
}
```

### 2.2 模板元编程（Template Metaprogramming）

```cpp
// C++11/14/17/20/23 通用
#include <iostream>
#include <type_traits>

// ============== 编译期计算 ==============

// 阶乘（递归模板）
template<int N>
struct Factorial {
    static constexpr int value = N * Factorial<N - 1>::value;
};

template<>
struct Factorial<0> {
    static constexpr int value = 1;
};

// C++14起可用变量模板
template<int N>
inline constexpr int factorial_v = Factorial<N>::value;

// 斐波那契数列
template<int N>
struct Fibonacci {
    static constexpr int value = Fibonacci<N - 1>::value + Fibonacci<N - 2>::value;
};

template<>
struct Fibonacci<0> {
    static constexpr int value = 0;
};

template<>
struct Fibonacci<1> {
    static constexpr int value = 1;
};

// ============== 类型列表（Type List）=============
template<typename... Types>
struct TypeList {
    static constexpr std::size_t size = sizeof...(Types);
};

// 获取列表头部
template<typename List>
struct Head;

template<typename First, typename... Rest>
struct Head<TypeList<First, Rest...>> {
    using type = First;
};

// 获取列表尾部
template<typename List>
struct Tail;

template<typename First, typename... Rest>
struct Tail<TypeList<First, Rest...>> {
    using type = TypeList<Rest...>;
};

// 在列表中查找类型
template<typename List, typename T>
struct Contains : std::false_type {};

template<typename First, typename... Rest, typename T>
struct Contains<TypeList<First, Rest...>, T> 
    : std::conditional_t<std::is_same_v<First, T>,
                         std::true_type,
                         Contains<TypeList<Rest...>, T>> {};

// 获取指定索引的类型
template<typename List, std::size_t N>
struct At;

template<typename First, typename... Rest>
struct At<TypeList<First, Rest...>, 0> {
    using type = First;
};

template<typename First, typename... Rest, std::size_t N>
struct At<TypeList<First, Rest...>, N> {
    using type = typename At<TypeList<Rest...>, N - 1>::type;
};

// 追加类型到列表
template<typename List, typename T>
struct Append;

template<typename... Types, typename T>
struct Append<TypeList<Types...>, T> {
    using type = TypeList<Types..., T>;
};

// 反转列表
template<typename List>
struct Reverse;

template<>
struct Reverse<TypeList<>> {
    using type = TypeList<>;
};

template<typename First, typename... Rest>
struct Reverse<TypeList<First, Rest...>> {
    using tail_reversed = typename Reverse<TypeList<Rest...>>::type;
    using type = typename Append<tail_reversed, First>::type;
};

// ============== 编译期条件 ==============
template<bool Condition, typename TrueType, typename FalseType>
struct If {
    using type = TrueType;
};

template<typename TrueType, typename FalseType>
struct If<false, TrueType, FalseType> {
    using type = FalseType;
};

// 编译期选择
template<int N>
using SelectType = typename If<(N > 10), int, double>::type;

// ============== 编译期循环（使用递归）=============

// 编译期整数序列
template<std::size_t... Is>
struct IndexSequence {};

// 生成整数序列
template<std::size_t N, std::size_t... Is>
struct MakeIndexSequenceImpl 
    : MakeIndexSequenceImpl<N - 1, N - 1, Is...> {};

template<std::size_t... Is>
struct MakeIndexSequenceImpl<0, Is...> {
    using type = IndexSequence<Is...>;
};

template<std::size_t N>
using MakeIndexSequence = typename MakeIndexSequenceImpl<N>::type;

// C++14起标准库提供std::make_index_sequence

// ============== 编译期排序（选择排序）=============
template<int... Values>
struct IntList;

template<int First, int... Rest>
struct IntList<First, Rest...> {
    static constexpr int head = First;
    using tail = IntList<Rest...>;
    static constexpr std::size_t size = 1 + sizeof...(Rest);
};

template<>
struct IntList<> {
    static constexpr std::size_t size = 0;
};

// 查找最小值
template<typename List, int Min = 2147483647>
struct MinValue;

template<int Min>
struct MinValue<IntList<>, Min> {
    static constexpr int value = Min;
};

template<int First, int... Rest, int Min>
struct MinValue<IntList<First, Rest...>, Min> {
    static constexpr int new_min = First < Min ? First : Min;
    static constexpr int value = MinValue<IntList<Rest...>, new_min>::value;
};

// 从列表中移除指定值（第一个匹配）
template<typename List, int Value>
struct RemoveFirst;

template<int Value>
struct RemoveFirst<IntList<>, Value> {
    using type = IntList<>;
};

template<int First, int... Rest, int Value>
struct RemoveFirst<IntList<First, Rest...>, Value> {
    using type = std::conditional_t<First == Value,
                                    IntList<Rest...>,
                                    typename Prepend<typename RemoveFirst<IntList<Rest...>, Value>::type, First>::type>;
};

template<typename List, int Value>
struct Prepend;

template<int... Values, int Value>
struct Prepend<IntList<Values...>, Value> {
    using type = IntList<Value, Values...>;
};

// 选择排序
template<typename List>
struct Sort;

template<>
struct Sort<IntList<>> {
    using type = IntList<>;
};

template<int... Values>
struct Sort<IntList<Values...>> {
    using list = IntList<Values...>;
    static constexpr int min = MinValue<list>::value;
    using rest = typename RemoveFirst<list, min>::type;
    using sorted_rest = typename Sort<rest>::type;
    using type = typename Prepend<sorted_rest, min>::type;
};

// ============== 测试 ==============
int main() {
    std::cout << "=== 编译期计算 ===" << std::endl;
    std::cout << "Factorial<5> = " << Factorial<5>::value << std::endl;
    std::cout << "Fibonacci<10> = " << Fibonacci<10>::value << std::endl;
    
    std::cout << "\n=== 类型列表 ===" << std::endl;
    using MyTypes = TypeList<int, double, char, float>;
    std::cout << "TypeList size = " << MyTypes::size << std::endl;
    
    using FirstType = Head<MyTypes>::type;
    std::cout << "First type is int: " << std::is_same_v<FirstType, int> << std::endl;
    
    std::cout << "Contains<double>: " << Contains<MyTypes, double>::value << std::endl;
    std::cout << "Contains<long>: " << Contains<MyTypes, long>::value << std::endl;
    
    using ThirdType = At<MyTypes, 2>::type;
    std::cout << "At<2> is char: " << std::is_same_v<ThirdType, char> << std::endl;
    
    using Reversed = Reverse<MyTypes>::type;
    std::cout << "Reversed[0] is float: " << std::is_same_v<At<Reversed, 0>::type, float> << std::endl;
    
    std::cout << "\n=== 编译期排序 ===" << std::endl;
    using Unsorted = IntList<5, 2, 8, 1, 9, 3>;
    using Sorted = Sort<Unsorted>::type;
    std::cout << "Sorted list created (compile-time)" << std::endl;
    
    return 0;
}
```

### 2.3 表达式模板（Expression Templates）

```cpp
// C++17 完整示例
#include <iostream>
#include <vector>
#include <cassert>

// ============== 简单向量类 ==============
class Vector {
public:
    Vector(std::size_t n) : data_(n) {}
    
    Vector(std::initializer_list<double> init) : data_(init) {}
    
    std::size_t size() const { return data_.size(); }
    
    double operator[](std::size_t i) const { return data_[i]; }
    double& operator[](std::size_t i) { return data_[i]; }
    
private:
    std::vector<double> data_;
};

// ============== 传统实现（有临时对象）=============
// Vector operator+(const Vector& a, const Vector& b) {
//     Vector result(a.size());
//     for (std::size_t i = 0; i < a.size(); ++i) {
//         result[i] = a[i] + b[i];
//     }
//     return result;  // 返回临时对象
// }
// 
// 问题：a + b + c 产生多个临时对象
// (a + b) 产生临时对象t1
// t1 + c 产生临时对象t2

// ============== 表达式模板实现 ==============

// 表达式基类
template<typename Expr>
class Expression {
public:
    double operator[](std::size_t i) const {
        return static_cast<const Expr&>(*this)[i];
    }
    
    std::size_t size() const {
        return static_cast<const Expr&>(*this).size();
    }
};

// 向量包装为表达式
class VectorExpr : public Expression<VectorExpr> {
public:
    VectorExpr(const Vector& vec) : vec_(vec) {}
    
    double operator[](std::size_t i) const { return vec_[i]; }
    std::size_t size() const { return vec_.size(); }
    
private:
    const Vector& vec_;
};

// 加法表达式
template<typename LHS, typename RHS>
class AddExpr : public Expression<AddExpr<LHS, RHS>> {
public:
    AddExpr(const LHS& lhs, const RHS& rhs) : lhs_(lhs), rhs_(rhs) {
        assert(lhs.size() == rhs.size());
    }
    
    double operator[](std::size_t i) const {
        return lhs_[i] + rhs_[i];  // 延迟计算
    }
    
    std::size_t size() const { return lhs_.size(); }
    
private:
    const LHS& lhs_;
    const RHS& rhs_;
};

// 减法表达式
template<typename LHS, typename RHS>
class SubExpr : public Expression<SubExpr<LHS, RHS>> {
public:
    SubExpr(const LHS& lhs, const RHS& rhs) : lhs_(lhs), rhs_(rhs) {
        assert(lhs.size() == rhs.size());
    }
    
    double operator[](std::size_t i) const {
        return lhs_[i] - rhs_[i];
    }
    
    std::size_t size() const { return lhs_.size(); }
    
private:
    const LHS& lhs_;
    const RHS& rhs_;
};

// 数乘表达式
template<typename Expr>
class ScaleExpr : public Expression<ScaleExpr<Expr>> {
public:
    ScaleExpr(double scalar, const Expr& expr) : scalar_(scalar), expr_(expr) {}
    
    double operator[](std::size_t i) const {
        return scalar_ * expr_[i];
    }
    
    std::size_t size() const { return expr_.size(); }
    
private:
    double scalar_;
    const Expr& expr_;
};

// 运算符重载
template<typename LHS, typename RHS>
AddExpr<LHS, RHS> operator+(const Expression<LHS>& lhs, const Expression<RHS>& rhs) {
    return AddExpr<LHS, RHS>(static_cast<const LHS&>(lhs), static_cast<const RHS&>(rhs));
}

template<typename LHS, typename RHS>
SubExpr<LHS, RHS> operator-(const Expression<LHS>& lhs, const Expression<RHS>& rhs) {
    return SubExpr<LHS, RHS>(static_cast<const LHS&>(lhs), static_cast<const RHS&>(rhs));
}

template<typename Expr>
ScaleExpr<Expr> operator*(double scalar, const Expression<Expr>& expr) {
    return ScaleExpr<Expr>(scalar, static_cast<const Expr&>(expr));
}

// Vector到表达式的隐式转换
VectorExpr expr(const Vector& vec) {
    return VectorExpr(vec);
}

// 赋值操作：实际计算发生在这里
class Vector;

// 前向声明
namespace detail {
    template<typename Expr>
    void assign(Vector& vec, const Expression<Expr>& expr);
}

class Vector {
public:
    Vector(std::size_t n) : data_(n) {}
    Vector(std::initializer_list<double> init) : data_(init) {}
    
    std::size_t size() const { return data_.size(); }
    double operator[](std::size_t i) const { return data_[i]; }
    double& operator[](std::size_t i) { return data_[i]; }
    
    // 从表达式赋值
    template<typename Expr>
    Vector& operator=(const Expression<Expr>& expr) {
        const Expr& e = static_cast<const Expr&>(expr);
        assert(size() == e.size());
        for (std::size_t i = 0; i < size(); ++i) {
            data_[i] = e[i];  // 这里才真正计算
        }
        return *this;
    }
    
private:
    std::vector<double> data_;
};

// ============== 测试 ==============
int main() {
    Vector a{1, 2, 3, 4, 5};
    Vector b{5, 4, 3, 2, 1};
    Vector c{1, 1, 1, 1, 1};
    
    Vector result(5);
    
    std::cout << "=== 传统方式（多次临时对象） ===" << std::endl;
    // result = a + b + c;  // 传统方式产生临时对象
    
    std::cout << "\n=== 表达式模板（延迟计算） ===" << std::endl;
    // 这里不计算，只构建表达式树
    auto expr_tree = expr(a) + expr(b) + expr(c);
    
    // 这里才真正计算
    result = expr_tree;
    
    std::cout << "result = a + b + c: ";
    for (std::size_t i = 0; i < result.size(); ++i) {
        std::cout << result[i] << " ";
    }
    std::cout << std::endl;
    
    // 复杂表达式
    std::cout << "\n=== 复杂表达式 2*a + 3*b - c ===" << std::endl;
    result = 2.0 * expr(a) + 3.0 * expr(b) - expr(c);
    
    std::cout << "result: ";
    for (std::size_t i = 0; i < result.size(); ++i) {
        std::cout << result[i] << " ";
    }
    std::cout << std::endl;
    
    return 0;
}
```

### 2.4 Tag Dispatch 模式

```cpp
// C++11/14/17/20/23 通用
#include <iostream>
#include <vector>
#include <list>
#include <type_traits>
#include <algorithm>

// ============== Tag类型定义 ==============
struct random_access_iterator_tag {};
struct bidirectional_iterator_tag {};
struct forward_iterator_tag {};
struct input_iterator_tag {};

// ============== 迭代器特征 ==============
template<typename Iterator>
struct iterator_traits {
    using iterator_category = typename Iterator::iterator_category;
};

// 指针特化
template<typename T>
struct iterator_traits<T*> {
    using iterator_category = random_access_iterator_tag;
};

// ============== Tag Dispatch实现 ==============

// 主函数：根据迭代器类型分发
template<typename Iterator>
void advance_impl(Iterator& it, int n, random_access_iterator_tag) {
    std::cout << "使用随机访问迭代器: O(1)" << std::endl;
    it += n;
}

template<typename Iterator>
void advance_impl(Iterator& it, int n, bidirectional_iterator_tag) {
    std::cout << "使用双向迭代器: O(n)" << std::endl;
    if (n >= 0) {
        while (n--) ++it;
    } else {
        while (n++) --it;
    }
}

template<typename Iterator>
void advance_impl(Iterator& it, int n, forward_iterator_tag) {
    std::cout << "使用前向迭代器: O(n)" << std::endl;
    while (n--) ++it;
}

// 包装函数
template<typename Iterator>
void my_advance(Iterator& it, int n) {
    using category = typename iterator_traits<Iterator>::iterator_category;
    advance_impl(it, n, category{});
}

// ============== 距离计算 ==============
template<typename Iterator>
int distance_impl(Iterator first, Iterator last, random_access_iterator_tag) {
    std::cout << "随机访问距离: O(1)" << std::endl;
    return last - first;
}

template<typename Iterator>
int distance_impl(Iterator first, Iterator last, forward_iterator_tag) {
    std::cout << "前向迭代器距离: O(n)" << std::endl;
    int count = 0;
    while (first != last) {
        ++first;
        ++count;
    }
    return count;
}

template<typename Iterator>
int my_distance(Iterator first, Iterator last) {
    using category = typename iterator_traits<Iterator>::iterator_category;
    return distance_impl(first, last, category{});
}

// ============== 自定义容器和迭代器 ==============
template<typename T>
class SimpleVector {
public:
    using iterator = T*;
    using iterator_category = random_access_iterator_tag;
    
    iterator begin() { return data_; }
    iterator end() { return data_ + size_; }
    
private:
    T data_[100];
    std::size_t size_ = 0;
};

template<typename T>
class SimpleList {
public:
    struct Node {
        T data;
        Node* next;
    };
    
    struct iterator {
        Node* node;
        using iterator_category = forward_iterator_tag;
        
        iterator& operator++() { node = node->next; return *this; }
        bool operator!=(const iterator& other) const { return node != other.node; }
        T& operator*() { return node->data; }
    };
    
    iterator begin() { return iterator{head_}; }
    iterator end() { return iterator{nullptr}; }
    
private:
    Node* head_ = nullptr;
};

// ============== 现代C++方式（C++17）=============
// 使用if constexpr替代tag dispatch
template<typename Iterator>
void modern_advance(Iterator& it, int n) {
    using category = typename iterator_traits<Iterator>::iterator_category;
    
    if constexpr (std::is_same_v<category, random_access_iterator_tag>) {
        std::cout << "现代方式-随机访问: O(1)" << std::endl;
        it += n;
    } else if constexpr (std::is_same_v<category, bidirectional_iterator_tag>) {
        std::cout << "现代方式-双向: O(n)" << std::endl;
        if (n >= 0) while (n--) ++it;
        else while (n++) --it;
    } else {
        std::cout << "现代方式-前向: O(n)" << std::endl;
        while (n--) ++it;
    }
}

// ============== 测试 ==============
int main() {
    std::cout << "=== Tag Dispatch ===" << std::endl;
    
    std::vector<int> vec{1, 2, 3, 4, 5};
    auto vec_it = vec.begin();
    my_advance(vec_it, 3);
    std::cout << "*vec_it = " << *vec_it << std::endl;
    
    std::list<int> lst{1, 2, 3, 4, 5};
    auto lst_it = lst.begin();
    my_advance(lst_it, 3);
    std::cout << "*lst_it = " << *lst_it << std::endl;
    
    std::cout << "\n=== Distance ===" << std::endl;
    std::cout << "vec distance: " << my_distance(vec.begin(), vec.end()) << std::endl;
    std::cout << "lst distance: " << my_distance(lst.begin(), lst.end()) << std::endl;
    
    std::cout << "\n=== Modern if constexpr ===" << std::endl;
    auto vec_it2 = vec.begin();
    modern_advance(vec_it2, 2);
    std::cout << "*vec_it2 = " << *vec_it2 << std::endl;
```

```cpp
// 创建策略
template<typename T>
struct HeapAllocation {
    static T* create() {
        std::cout << "Heap allocation" << std::endl;
        return new T();
    }
    
    static void destroy(T* ptr) {
        std::cout << "Heap deallocation" << std::endl;
        delete ptr;
    }
};

template<typename T>
struct StackAllocation {
    static T* create() {
        std::cout << "Stack allocation (simulated)" << std::endl;
        // 实际使用placement new在预分配内存上构造
        static alignas(alignof(T)) char buffer[sizeof(T)];
        return reinterpret_cast<T*>(buffer);
    }
    
    static void destroy(T* ptr) {
        std::cout << "Stack deallocation (simulated)" << std::endl;
        ptr->~T();
    }
};

// 线程安全策略
struct SingleThreaded {
    void lock() {}
    void unlock() {}
};

struct MultiThreaded {
    void lock() { std::cout << "Lock acquired" << std::endl; }
    void unlock() { std::cout << "Lock released" << std::endl; }
};

// 验证策略
template<typename T>
struct NoValidation {
    static void validate(const T&) {}
};

template<typename T>
struct RangeValidation {
    static void validate(const T& val) {
        if (val < 0 || val > 100) {
            throw std::out_of_range("Value out of range [0, 100]");
        }
    }
};

// ============== 使用策略的类 ==============
template<typename T,
         template<typename> class AllocationPolicy = HeapAllocation,
         typename ThreadingPolicy = SingleThreaded,
         template<typename> class ValidationPolicy = NoValidation>
class SmartValue {
public:
    using AllocPolicy = AllocationPolicy<T>;
    using ValidPolicy = ValidationPolicy<T>;
    
    SmartValue() {
        threading_.lock();
        ptr_ = AllocPolicy::create();
        threading_.unlock();
    }
    
    explicit SmartValue(const T& val) : SmartValue() {
        ValidPolicy::validate(val);
        *ptr_ = val;
    }
    
    ~SmartValue() {
        threading_.lock();
        AllocPolicy::destroy(ptr_);
        threading_.unlock();
    }
    
    void set(const T& val) {
        ValidPolicy::validate(val);
        threading_.lock();
        *ptr_ = val;
        threading_.unlock();
    }
    
    T get() const {
        threading_.lock();
        T val = *ptr_;
        threading_.unlock();
        return val;
    }
    
private:
    T* ptr_;
    ThreadingPolicy threading_;
};

// ============== 输出策略 ==============

    static void write(const std::string& msg) {
        std::cout << "[Console] " << msg << std::endl;
    }
};

    static void write(const std::string& msg) {
        std::cout << "[File] " << msg << std::endl;  // 模拟文件输出
    }
};

    static void write(const std::string& msg) {
        std::cout << "[Network] " << msg << std::endl;  // 模拟网络输出
    }
};

// ============== 日志类 ==============
template<typename OutputPolicy = ConsoleOutput>
class Logger {
public:
    void log(const std::string& msg) {
        OutputPolicy::write(msg);
    }
    
    template<typename... Args>
    void logf(const char* fmt, Args... args) {
        char buffer[256];
        snprintf(buffer, sizeof(buffer), fmt, args...);
        OutputPolicy::write(buffer);
    }
};

// ============== 组合多个策略 ==============
template<typename T, typename... Policies>
class MultiPolicyContainer;

class MultiPolicyContainer<T> {
public:
    void apply(const T& val) {}

    : private FirstPolicy, private MultiPolicyContainer<T, RestPolicies...> {
public:
    void apply(const T& val) {
        FirstPolicy::apply(val);
        MultiPolicyContainer<T, RestPolicies...>::apply(val);
    }
```

### 3.1 完整示例：编译期状态机

```cpp
// C++17 完整示例
#include <iostream>
#include <type_traits>

// ============== 状态定义 ==============
struct StateInit {};
struct StateRunning {};
struct StatePaused {};
struct StateStopped {};

// ============== 事件定义 ==============
struct EventStart {};
struct EventPause {};
struct EventResume {};
struct EventStop {};

// ============== 状态转换表（编译期）=============
// 使用类型列表定义状态机

template<typename State, typename Event>
struct Transition;

// 定义转换规则
template<> struct Transition<StateInit, EventStart> { using type = StateRunning; };
template<> struct Transition<StateRunning, EventPause> { using type = StatePaused; };
template<> struct Transition<StateRunning, EventStop> { using type = StateStopped; };
template<> struct Transition<StatePaused, EventResume> { using type = StateRunning; };
template<> struct Transition<StatePaused, EventStop> { using type = StateStopped; };

// 默认：状态不变
template<typename State, typename Event>
struct Transition {
    using type = State;
};

// ============== 状态机实现 ==============
template<typename CurrentState>
class StateMachine {
public:
    using state = CurrentState;
    
    template<typename Event>
    auto process_event(Event) {
        using NextState = typename Transition<CurrentState, Event>::type;
        print_transition<CurrentState, Event, NextState>();
        return StateMachine<NextState>{};
    }
    
    void print_current_state() const {
        print_state_name<CurrentState>();
    }
    
private:
    template<typename S>
    static void print_state_name() {
        if constexpr (std::is_same_v<S, StateInit>) {
            std::cout << "Init";
        } else if constexpr (std::is_same_v<S, StateRunning>) {
            std::cout << "Running";
        } else if constexpr (std::is_same_v<S, StatePaused>) {
            std::cout << "Paused";
        } else if constexpr (std::is_same_v<S, StateStopped>) {
            std::cout << "Stopped";
        }
    }
    
    template<typename From, typename Event, typename To>
    static void print_transition() {
        std::cout << "Transition: ";
        print_state_name<From>();
        std::cout << " -> ";
        print_state_name<To>();
        std::cout << std::endl;
    }
};

// ============== 测试 ==============
int main() {
    std::cout << "=== 编译期状态机 ===" << std::endl;
    
    StateMachine<StateInit> sm;
    sm.print_current_state();
    std::cout << std::endl;
    
    auto sm1 = sm.process_event(EventStart{});
    auto sm2 = sm1.process_event(EventPause{});
    auto sm3 = sm2.process_event(EventResume{});
    auto sm4 = sm3.process_event(EventStop{});
    
    std::cout << "\n最终状态: ";
    sm4.print_current_state();
    std::cout << std::endl;
    
    // 无效转换测试
    std::cout << "\n=== 无效转换（状态不变） ===" << std::endl;
    StateMachine<StateInit> sm_init;
    auto sm_same = sm_init.process_event(EventStop{});  // 无效事件
    std::cout << "状态: ";
    sm_same.print_current_state();
    std::cout << std::endl;
    
    return 0;
}
```

```cpp
};

// 优势：
// 1. 编译期组合，零运行时开销
// 2. 灵活定制，不产生类爆炸
// 3. 策略之间相互独立
```

#### Q5: 表达式模板如何避免临时对象？

**标准答案**：

```cpp
// 传统方式：产生临时对象
Vector operator+(const Vector& a, const Vector& b);
// a + b + c 产生2个临时对象

// 表达式模板：延迟计算
template<typename LHS, typename RHS>
class AddExpr {
public:
    double operator[](size_t i) const {
        return lhs_[i] + rhs_[i];  // 计算延迟到这里
    }
};

// a + b + c 构建表达式树，不计算
// 赋值时才计算：result = a + b + c
```

## 5. 最佳实践

### 5.1 CRTP使用建议

```cpp
// 1. 使用静态断言确保正确使用
template<typename Derived>
class CRTPBase {
public:
    CRTPBase() {
        static_assert(std::is_base_of_v<CRTPBase, Derived>,
                      "Derived must inherit from CRTPBase<Derived>");
    }
};

// 2. 提供默认实现
template<typename Derived>
class Interface {
public:
    void algorithm() {
        // 默认实现或强制派生类实现
        static_cast<Derived*>(this)->implementation();
    }
};

// 3. 使用friend访问派生类私有成员
template<typename Derived>
class Base {
    friend Derived;  // 允许派生类访问基类私有成员
};
```

### 5.2 Concepts使用建议

```cpp
// 1. 命名概念使用大驼峰
template<typename T>
concept Container = /* ... */;

// 2. 组合标准概念
template<typename T>
concept Numeric = std::integral<T> || std::floating_point<T>;

// 3. 使用requires子句表达复杂约束
template<typename T>
    requires std::copyable<T> && std::equality_comparable<T>
class Buffer { };

// 4. 为概念提供简写
template<std::integral T>
void func(T);  // 简写形式
```

## 6. 常见陷阱与UB

### 6.1 CRTP陷阱

```cpp
// 陷阱1：派生类未完全定义时访问
template<typename Derived>
class Base {
public:
    Base() {
        // 危险：Derived可能还未完全定义
        static_cast<Derived*>(this)->foo();  // UB！
    }
};

// 陷阱2：多重继承CRTP冲突
class A : public Base<A> {};
class B : public Base<B> {};
class C : public A, public B {};  // 如果有共同基类会出问题

// 陷阱3：忘记将派生类设为友元
template<typename Derived>
class Counter {
    static int count;  // 私有
};
// 派生类无法访问count，除非设为友元
```

### 6.2 模板元编程陷阱

```cpp
// 陷阱1：递归深度限制
template<int N>
struct Deep {
    static constexpr int value = Deep<N - 1>::value;
};
// Deep<10000> 可能导致编译器栈溢出

// 陷阱2：不完全类型实例化
template<typename T>
struct Check {
    static constexpr bool value = sizeof(T) > 0;
};
// Check<IncompleteType> 是UB

// 陷阱3：模板递归中的副作用
template<int N>
struct SideEffect {
    static constexpr int value = (global_var++, N);  // UB！
};
```
