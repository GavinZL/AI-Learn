# 智能指针深入详细解析

> **核心结论**：智能指针是 RAII 原则在内存管理上的应用。`std::unique_ptr` 表示独占所有权，`std::shared_ptr` 通过引用计数实现共享所有权，`std::weak_ptr` 用于打破循环引用。理解控制块结构是掌握 shared_ptr 线程安全性的关键。

---

## 1. Why — 为什么需要智能指针

### 1.1 裸指针的问题

```cpp
// 裸指针的典型问题
void rawPointerProblems() {
    // 问题 1：内存泄漏
    int* p = new int(42);
    if (someCondition()) {
        return;  // 泄漏！未 delete
    }
    delete p;
    
    // 问题 2：重复释放
    int* q = new int(42);
    delete q;
    delete q;  // UB！重复释放
    
    // 问题 3：使用已释放内存（use-after-free）
    int* r = new int(42);
    delete r;
    *r = 100;  // UB！
    
    // 问题 4：异常安全
    void* resource = acquireResource();
    doWork();  // 如果抛出异常，resource 泄漏
    releaseResource(resource);
}
```

### 1.2 智能指针的价值

```cpp
#include <memory>

void smartPointerSolution() {
    // 自动内存管理
    auto p = std::make_unique<int>(42);
    if (someCondition()) {
        return;  // 自动释放，无泄漏
    }
    // 作用域结束时自动释放
    
    // 异常安全
    auto resource = std::make_unique<Resource>();
    doWork();  // 即使抛出异常，resource 也会被释放
}  // 自动调用析构函数
```

## 2. std::unique_ptr — 独占所有权

### 2.1 基本用法

```cpp
#include <memory>
#include <iostream>

void uniquePtrBasic() {
    // 创建 unique_ptr
    std::unique_ptr<int> p1(new int(42));  // 直接构造
    auto p2 = std::make_unique<int>(42);   // C++14 推荐（异常安全）
    
    // 访问
    std::cout << *p1 << std::endl;  // 解引用
    std::cout << p1.get() << std::endl;  // 获取裸指针
    
    // 所有权转移
    std::unique_ptr<int> p3 = std::move(p1);  // p1 置空
    // p1 现在是 nullptr
    
    // 释放所有权
    int* raw = p2.release();  // 释放所有权，返回裸指针
    delete raw;  // 必须手动释放
    
    // 重置
    p3.reset();  // 释放资源，置为 nullptr
    p3.reset(new int(100));  // 释放旧资源，指向新资源
    
    // 检查
    if (p3) {  // operator bool
        std::cout << "p3 owns resource" << std::endl;
    }
}
```

### 2.4 unique_ptr 的实现原理

```cpp
// 简化版实现（概念上）
template<typename T, typename Deleter = std::default_delete<T>>
class unique_ptr {
    T* ptr_;
    Deleter deleter_;
    
public:
    explicit unique_ptr(T* ptr = nullptr) : ptr_(ptr) {}
    
    ~unique_ptr() {
        if (ptr_) {
            deleter_(ptr_);
        }
    }
    
    // 禁止拷贝
    unique_ptr(const unique_ptr&) = delete;
    unique_ptr& operator=(const unique_ptr&) = delete;
    
    // 支持移动
    unique_ptr(unique_ptr&& other) noexcept 
        : ptr_(other.ptr_), deleter_(std::move(other.deleter_)) {
        other.ptr_ = nullptr;
    }
    
    unique_ptr& operator=(unique_ptr&& other) noexcept {
        if (this != &other) {
            reset();
            ptr_ = other.ptr_;
            deleter_ = std::move(other.deleter_);
            other.ptr_ = nullptr;
        }
        return *this;
    }
    
    T& operator*() const { return *ptr_; }
    T* operator->() const { return ptr_; }
    T* get() const { return ptr_; }
    
    T* release() {
        T* tmp = ptr_;
        ptr_ = nullptr;
        return tmp;
    }
    
    void reset(T* ptr = nullptr) {
        if (ptr_) {
            deleter_(ptr_);
        }
        ptr_ = ptr;
    }
    
    explicit operator bool() const { return ptr_ != nullptr; }
};
```

## 3. std::shared_ptr — 共享所有权

### 3.1 基本用法

```cpp
#include <memory>
#include <iostream>

void sharedPtrBasic() {
    // 创建 shared_ptr
    std::shared_ptr<int> p1(new int(42));  // 直接构造
    auto p2 = std::make_shared<int>(42);   // 推荐（更高效）
    
    // 共享所有权
    std::shared_ptr<int> p3 = p1;  // 引用计数 +1
    std::cout << p1.use_count() << std::endl;  // 3
    
    // 访问
    std::cout << *p1 << std::endl;
    std::cout << p1.get() << std::endl;
    
    // 重置
    p1.reset();  // 引用计数 -1
    std::cout << p3.use_count() << std::endl;  // 2
    
    // 检查唯一所有权
    if (p2.unique()) {
        std::cout << "Only owner" << std::endl;
    }
}
```

### 3.2 控制块结构

```cpp
// shared_ptr 内部结构（概念上）
// 
// 控制块（Control Block）          shared_ptr
// ┌─────────────────────┐         ┌──────────┐
// │ 引用计数（强引用）    │         │ 对象指针  │
// │ shared_count        │◄────────│ ptr_     │
// ├─────────────────────┤         ├──────────┤
// │ 弱引用计数          │         │ 控制块指针 │
// │ weak_count          │◄────────│ ctrl_    │
// ├─────────────────────┤         └──────────┘
// │ 删除器              │
// │ deleter             │
// ├─────────────────────┤
// │ 分配器              │
// │ allocator           │
// ├─────────────────────┤
// │ 指向的对象          │
// │ T object            │
// └─────────────────────┘

// 引用计数规则：
// - shared_count == 0：删除对象
// - shared_count == 0 && weak_count == 0：删除控制块
```

### 3.3 make_shared vs new + shared_ptr

```cpp
#include <memory>
#include <iostream>

class Widget {
public:
    Widget() { std::cout << "Widget constructed" << std::endl; }
    ~Widget() { std::cout << "Widget destroyed" << std::endl; }
};

void makeSharedComparison() {
    // 方法 1：new + shared_ptr（两次分配）
    // 1. new Widget() - 分配对象
    // 2. new ControlBlock() - 分配控制块
    std::shared_ptr<Widget> p1(new Widget());
    
    // 方法 2：make_shared（一次分配）
    // 分配一块内存：| ControlBlock | Widget |
    auto p2 = std::make_shared<Widget>();
    
    // make_shared 的优势：
    // 1. 性能更好（一次分配，缓存友好）
    // 2. 异常安全
    // 
    // make_shared 的劣势：
    // 1. 控制块和对象内存绑定，weak_ptr 会延长对象内存生命周期
    // 2. 无法使用自定义删除器
}

// 异常安全对比
void exceptionSafety() {
    // 不安全：如果 new Widget 后，shared_ptr 构造前抛出异常，泄漏
    process(std::shared_ptr<Widget>(new Widget()), mayThrow());
    
    // 安全：make_shared 是原子操作
    process(std::make_shared<Widget>(), mayThrow());
}
```

### 3.5 shared_ptr 的线程安全性

```cpp
#include <memory>
#include <thread>
#include <vector>
#include <iostream>

// shared_ptr 的线程安全级别
void threadSafety() {
    auto ptr = std::make_shared<int>(42);
    
    // 线程 1：读取 shared_ptr
    std::thread t1([ptr]() {  // 拷贝 shared_ptr（线程安全）
        std::cout << *ptr << std::endl;
    });
    
    // 线程 2：读取 shared_ptr
    std::thread t2([ptr]() {
        std::cout << *ptr << std::endl;
    });
    
    t1.join();
    t2.join();
    
    // 结论：
    // 1. 引用计数的操作是线程安全的（原子操作）
    // 2. 对 shared_ptr 对象的读写（拷贝、赋值）是线程安全的
    // 3. 对指向的对象的访问不是线程安全的！
}

// 非线程安全的例子
void notThreadSafe() {
    auto ptr = std::make_shared<int>(0);
    
    std::vector<std::thread> threads;
    for (int i = 0; i < 10; ++i) {
        threads.emplace_back([ptr]() {
            for (int j = 0; j < 1000; ++j) {
                (*ptr)++;  // 数据竞争！不是线程安全的
            }
        });
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    std::cout << *ptr << std::endl;  // 结果不确定（可能小于 10000）
}

// 线程安全的访问
void threadSafeAccess() {
    auto ptr = std::make_shared<int>(0);
    std::mutex mtx;
    
    std::vector<std::thread> threads;
    for (int i = 0; i < 10; ++i) {
        threads.emplace_back([ptr, &mtx]() {
            for (int j = 0; j < 1000; ++j) {
                std::lock_guard<std::mutex> lock(mtx);
                (*ptr)++;
            }
        });
    }
    
    for (auto& t : threads) {
        t.join();
    }
    
    std::cout << *ptr << std::endl;  // 10000
}
```

## 4. std::weak_ptr — 弱引用

### 4.1 基本用法

```cpp
#include <memory>
#include <iostream>

void weakPtrBasic() {
    std::shared_ptr<int> shared = std::make_shared<int>(42);
    std::weak_ptr<int> weak = shared;  // 不增加引用计数
    
    std::cout << shared.use_count() << std::endl;  // 1
    
    // 访问：需要转换为 shared_ptr
    if (auto locked = weak.lock()) {  // 尝试获取 shared_ptr
        std::cout << *locked << std::endl;  // 42
        std::cout << shared.use_count() << std::endl;  // 2
    }  // locked 销毁，引用计数 -1
    
    // 检查是否过期
    if (!weak.expired()) {
        std::cout << "Resource still alive" << std::endl;
    }
    
    // 释放 shared_ptr
    shared.reset();
    
    if (weak.expired()) {
        std::cout << "Resource destroyed" << std::endl;
    }
    
    // lock() 返回空 shared_ptr
    auto locked = weak.lock();
    if (!locked) {
        std::cout << "Failed to lock" << std::endl;
    }
}
```

### 4.2 打破循环引用

```cpp
#include <memory>
#include <iostream>

// 循环引用问题
struct NodeBad {
    std::shared_ptr<NodeBad> next;
    ~NodeBad() { std::cout << "NodeBad destroyed" << std::endl; }
};

void circularReferenceProblem() {
    auto node1 = std::make_shared<NodeBad>();
    auto node2 = std::make_shared<NodeBad>();
    
    node1->next = node2;
    node2->next = node1;  // 循环引用！
    
    // 引用计数：node1: 2, node2: 2
}  // 内存泄漏！引用计数永远不会变为 0

// 使用 weak_ptr 解决
struct NodeGood {
    std::weak_ptr<NodeGood> next;  // 弱引用
    ~NodeGood() { std::cout << "NodeGood destroyed" << std::endl; }
};

void circularReferenceSolution() {
    auto node1 = std::make_shared<NodeGood>();
    auto node2 = std::make_shared<NodeGood>();
    
    node1->next = node2;  // weak_ptr，不增加引用计数
    node2->next = node1;  // weak_ptr
    
    // 引用计数：node1: 1, node2: 1
}  // 正常销毁
```

### 4.3 观察者模式实现

```cpp
#include <memory>
#include <vector>
#include <iostream>
#include <algorithm>

class Subject;

class Observer {
public:
    virtual ~Observer() = default;
    virtual void onNotify(Subject& subject) = 0;
};

class Subject {
    std::vector<std::weak_ptr<Observer>> observers_;
    int state_ = 0;
    
public:
    void addObserver(std::shared_ptr<Observer> observer) {
        observers_.push_back(observer);
    }
    
    void removeObserver(std::shared_ptr<Observer> observer) {
        observers_.erase(
            std::remove_if(observers_.begin(), observers_.end(),
                [&observer](const std::weak_ptr<Observer>& wp) {
                    return wp.expired() || wp.lock() == observer;
                }),
            observers_.end()
        );
    }
    
    void notify() {
        for (auto it = observers_.begin(); it != observers_.end();) {
            if (auto sp = it->lock()) {
                sp->onNotify(*this);
                ++it;
            } else {
                // 观察者已销毁，清理
                it = observers_.erase(it);
            }
        }
    }
    
    void setState(int state) {
        state_ = state;
        notify();
    }
    
    int getState() const { return state_; }
};

class ConcreteObserver : public Observer {
    std::string name_;
public:
    explicit ConcreteObserver(std::string name) : name_(std::move(name)) {}
    
    void onNotify(Subject& subject) override {
        std::cout << name_ << " received update: " << subject.getState() << std::endl;
    }
};

void observerPatternExample() {
    Subject subject;
    
    {
        auto observer1 = std::make_shared<ConcreteObserver>("Observer1");
        auto observer2 = std::make_shared<ConcreteObserver>("Observer2");
        
        subject.addObserver(observer1);
        subject.addObserver(observer2);
        
        subject.setState(42);  // 两个观察者都收到通知
    }  // 观察者在这里销毁
    
    // Subject 仍然存活，但观察者已自动清理
    subject.setState(100);  // 无通知（观察者已清理）
}
```

## 5. enable_shared_from_this

### 5.1 基本用法

```cpp
#include <memory>
#include <iostream>

// 问题：从 this 创建 shared_ptr
class BadExample {
public:
    std::shared_ptr<BadExample> getShared() {
        return std::shared_ptr<BadExample>(this);  // 错误！
    }
};

void badExample() {
    auto obj = std::make_shared<BadExample>();
    auto obj2 = obj->getShared();  // 创建新的控制块！
    
    // 现在有两个独立的 shared_ptr 管理同一个对象
    // 双重释放！
}

// 正确做法：继承 enable_shared_from_this
class GoodExample : public std::enable_shared_from_this<GoodExample> {
public:
    std::shared_ptr<GoodExample> getShared() {
        return shared_from_this();  // 使用已有的控制块
    }
    
    void doWork() {
        // 异步操作需要保持对象存活
        auto self = shared_from_this();
        std::thread([self]() {
            // 使用 self 安全地访问对象
            self->someMethod();
        }).detach();
    }
    
    void someMethod() {}
};

void goodExample() {
    auto obj = std::make_shared<GoodExample>();
    auto obj2 = obj->getShared();  // 共享控制块
    
    std::cout << obj.use_count() << std::endl;  // 2
}
```

### 5.2 实现原理

```cpp
// enable_shared_from_this 简化实现
template<typename T>
class enable_shared_from_this {
    mutable std::weak_ptr<T> weak_this_;
    
public:
    std::shared_ptr<T> shared_from_this() {
        return std::shared_ptr<T>(weak_this_);
    }
    
    std::shared_ptr<const T> shared_from_this() const {
        return std::shared_ptr<const T>(weak_this_);
    }
    
    // 由 shared_ptr 构造函数调用
    template<typename>
    friend class shared_ptr;
    
    void internal_accept_owner(const std::shared_ptr<T>& ptr) const {
        if (weak_this_.expired()) {
            weak_this_ = ptr;
        }
    }
};
```

### 6.2 类型转换

```cpp
#include <memory>
#include <iostream>

class Base {
public:
    virtual ~Base() = default;
};

class Derived : public Base {
public:
    void derivedOnly() { std::cout << "Derived method" << std::endl; }
};

void smartPointerCast() {
    std::shared_ptr<Base> base = std::make_shared<Derived>();
    
    // static_pointer_cast：上行转换（安全）
    std::shared_ptr<Derived> derived1 = 
        std::static_pointer_cast<Derived>(base);
    derived1->derivedOnly();
    
    // dynamic_pointer_cast：安全下行转换
    std::shared_ptr<Base> base2 = std::make_shared<Base>();
    if (auto derived2 = std::dynamic_pointer_cast<Derived>(base2)) {
        derived2->derivedOnly();  // 不会执行
    } else {
        std::cout << "Conversion failed" << std::endl;
    }
    
    // const_pointer_cast：const 转换
    std::shared_ptr<const Derived> constPtr = std::make_shared<Derived>();
    std::shared_ptr<Derived> mutablePtr = 
        std::const_pointer_cast<Derived>(constPtr);
}
```

## 7. 性能对比

### 7.1 开销分析

| 特性 | unique_ptr | shared_ptr | weak_ptr |
|------|-----------|-----------|----------|
| 大小 | sizeof(T*) | 2 * sizeof(void*) | sizeof(void*) |
| 拷贝 | 禁止 | 原子操作 +1 | 原子操作 +1 |
| 移动 | 指针交换 | 指针交换 | 指针交换 |
| 析构 | 直接 delete | 原子 -1，可能 delete | 原子 -1 |
| 线程安全 | N/A | 引用计数安全 | 引用计数安全 |

### 7.2 性能测试

```cpp
void performanceTest() {

    // benchmark<std::unique_ptr<int>>("unique_ptr", N);
    
    // shared_ptr 有引用计数开销
    benchmark<std::shared_ptr<int>>("shared_ptr", N);
}
```

## 8. 面试要点

### 8.1 高频面试题

#### Q1: make_shared 和 new + shared_ptr 的区别？

**答案**：
- **make_shared**：一次分配内存（对象 + 控制块），缓存友好，异常安全
- **new + shared_ptr**：两次分配，可能有异常安全问题
- **make_shared 劣势**：控制块和对象内存绑定，weak_ptr 会延长对象内存生命周期

```cpp
// 推荐
auto p1 = std::make_shared<Widget>();

// 不推荐
std::shared_ptr<Widget> p2(new Widget());
```

#### Q2: shared_ptr 的线程安全到什么程度？

**答案**：
- **引用计数操作**：线程安全（原子操作）
- **shared_ptr 对象本身**：拷贝、赋值是线程安全的
- **指向的对象**：不是线程安全的，需要额外同步

```cpp
std::shared_ptr<int> ptr = std::make_shared<int>(0);

// 线程安全：引用计数操作
std::thread([&]() { auto p = ptr; }).join();  // OK

// 不是线程安全：对象访问
std::thread([&]() { (*ptr)++; }).join();  // 数据竞争！
```

#### Q3: 如何检测 shared_ptr 的循环引用？

**答案**：
- **使用 weak_ptr**：将其中一个引用改为 weak_ptr
- **使用工具**：AddressSanitizer、Valgrind 检测内存泄漏
- **代码审查**：检查双向引用关系

```cpp
// 循环引用
struct A { std::shared_ptr<B> b; };
struct B { std::shared_ptr<A> a; };

// 解决方案
struct A { std::shared_ptr<B> b; };
struct B { std::weak_ptr<A> a; };  // 使用 weak_ptr
```

#### Q4: unique_ptr 可以放入容器吗？

**答案**：可以，但需要移动语义

```cpp
std::vector<std::unique_ptr<Widget>> vec;

vec.push_back(std::make_unique<Widget>());  // OK

auto ptr = std::make_unique<Widget>();
vec.push_back(std::move(ptr));  // 必须 move

// 访问
vec[0]->doWork();

// 遍历
for (auto& p : vec) {
    p->doWork();
}
```

#### Q5: enable_shared_from_this 的作用？

**答案**：
- 允许对象安全地获取自身的 shared_ptr
- 避免从 this 直接构造 shared_ptr 导致的双重控制块问题
- 必须在 shared_ptr 管理对象后才能调用 shared_from_this()

```cpp
class Widget : public std::enable_shared_from_this<Widget> {
public:
    std::shared_ptr<Widget> getShared() {
        return shared_from_this();  // 安全
    }
};
```

## 9. 最佳实践

### 9.1 选择正确的智能指针

```cpp
// 1. 默认使用 unique_ptr（零开销，表达所有权）
std::unique_ptr<Widget> createWidget() {
    return std::make_unique<Widget>();
}

// 2. 需要共享所有权时使用 shared_ptr
class Document {
    std::shared_ptr<Style> style_;  // 多个文档可能共享样式
};

// 3. 使用 weak_ptr 打破循环引用
struct Node {
    std::weak_ptr<Node> parent_;  // 弱引用父节点
    std::vector<std::shared_ptr<Node>> children_;
};

// 4. 避免裸指针，除非作为参数
void process(Widget* widget);  // 仅观察，不拥有
void process(std::unique_ptr<Widget> widget);  // 转移所有权
void process(std::shared_ptr<Widget> widget);  // 共享所有权
```

### 9.2 异常安全

```cpp
// 异常安全的资源管理
void exceptionSafe() {
    // 不好：可能泄漏
    Widget* w = new Widget();
    mayThrow();  // 如果抛出，w 泄漏
    delete w;
    
    // 好：RAII
    auto w = std::make_unique<Widget>();
    mayThrow();  // 安全，自动释放
}

// 构造函数异常安全
class SafeClass {
    std::unique_ptr<Resource> resource1_;
    std::unique_ptr<Resource> resource2_;
    
public:
    SafeClass() 
        : resource1_(std::make_unique<Resource>()),
          resource2_(std::make_unique<Resource>()) {}
    // 如果 resource2_ 构造失败，resource1_ 自动释放
};
```

## 10. 常见陷阱

### 10.1 裸指针与智能指针混用

```cpp
// 陷阱：从裸指针创建多个 shared_ptr
int* raw = new int(42);
std::shared_ptr<int> p1(raw);
std::shared_ptr<int> p2(raw);  // 错误！双重释放

// 正确做法
auto p1 = std::make_shared<int>(42);
std::shared_ptr<int> p2 = p1;  // 共享所有权
```

### 10.2 shared_ptr 的 this 问题

```cpp
// 陷阱：从 this 创建 shared_ptr
class Bad {
public:
    void register() {
        globalList.push_back(std::shared_ptr<Bad>(this));  // 错误！
    }
};

// 解决方案：继承 enable_shared_from_this
class Good : public std::enable_shared_from_this<Good> {
public:
    void register() {
        globalList.push_back(shared_from_this());  // 正确
    }
};
```

### 10.3 数组删除问题

```cpp
// 陷阱：使用默认删除器删除数组
std::shared_ptr<int> p(new int[10]);  // 错误！使用 delete 而非 delete[]

// 正确做法 1：使用数组特化
std::unique_ptr<int[]> arr(new int[10]);

// 正确做法 2：使用自定义删除器
std::shared_ptr<int> p2(new int[10], std::default_delete<int[]>());

// 正确做法 3：使用 vector（推荐）
std::vector<int> vec(10);
```

## 11. 总结

### 核心要点回顾

1. **unique_ptr**：独占所有权，零开销，优先使用
2. **shared_ptr**：共享所有权，引用计数，注意线程安全边界
3. **weak_ptr**：弱引用，打破循环引用，实现观察者模式
4. **make_shared/make_unique**：推荐创建方式，异常安全，性能更好
5. **enable_shared_from_this**：安全获取 this 的 shared_ptr

### 使用决策树

```
需要共享所有权？
├── 否 → 使用 unique_ptr
└── 是 → 需要打破循环引用？
    ├── 是 → 使用 weak_ptr
    └── 否 → 使用 shared_ptr
```

智能指针是现代 C++ 内存管理的基石，正确使用可以消除绝大多数内存安全问题。
