## 1. 时域降噪的基本思想

### 1.1 利用时间维度的冗余信息

视频或连续拍摄的图像序列中，相邻帧包含大量相似信息。噪声在各帧间是**独立随机**的，而信号是**相关一致**的：

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                        时域降噪基本原理                                      │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│   帧序列中的信号与噪声：                                                     │
│                                                                              │
│   Frame t-2:  Signal + Noise_1                                              │
│   Frame t-1:  Signal + Noise_2                                              │
│   Frame t  :  Signal + Noise_3    ← 当前帧                                  │
│   Frame t+1:  Signal + Noise_4                                              │
│   Frame t+2:  Signal + Noise_5                                              │
│                 ↓        ↓                                                   │
│               一致      独立                                                 │
│                                                                              │
│   多帧平均效果：                                                             │
│   Average = Signal + (Noise_1 + Noise_2 + ... + Noise_N) / N               │
│                        ↓                                                     │
│                   趋向于零（中心极限定理）                                   │
│                                                                              │
│   直观理解：                                                                 │
│   ┌─────────────────────────────────────────────────────┐                  │
│   │ 单帧（高噪声）     多帧平均（低噪声）                │                  │
│   │                                                      │                  │
│   │  ░▒█░▓░▒█▓░       ████████████                      │                  │
│   │  ▒░█▓░▒░▓▒░  →    ████████████                      │                  │
│   │  ░▓░▒█░▓░▒░       ████████████                      │                  │
│   │                                                      │                  │
│   └─────────────────────────────────────────────────────┘                  │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 1.2 多帧平均的信噪比提升（√N法则）

假设每帧噪声独立同分布，方差为 $\sigma^2$：

$$
\text{Var}\left(\frac{1}{N}\sum_{i=1}^{N} I_i\right) = \frac{\sigma^2}{N}
$$

信噪比提升：

$$
\text{SNR}_{avg} = \sqrt{N} \cdot \text{SNR}_{single}
$$

| 帧数N | SNR提升 | dB提升 |
|-------|---------|--------|
| 2 | 1.41× | +3 dB |
| 4 | 2× | +6 dB |
| 8 | 2.83× | +9 dB |
| 16 | 4× | +12 dB |

## 2. 多帧降噪（MFNR）

### 2.1 MFNR概述

Multi-Frame Noise Reduction是手机夜景模式的核心技术，通过拍摄多张短曝光图像并融合，达到长曝光效果同时避免运动模糊。

### 2.2 帧对齐技术

#### 全局仿射对齐

适用于相机运动，假设变换为仿射变换：

$$
\begin{bmatrix} x' \\ y' \end{bmatrix} = 
\begin{bmatrix} a & b \\ c & d \end{bmatrix}
\begin{bmatrix} x \\ y \end{bmatrix} +
\begin{bmatrix} t_x \\ t_y \end{bmatrix}
$$

```python
import cv2
import numpy as np

def align_frames_affine(reference, target):
    """
    全局仿射对齐
    """
    # 转灰度
    ref_gray = cv2.cvtColor(reference, cv2.COLOR_BGR2GRAY)
    tgt_gray = cv2.cvtColor(target, cv2.COLOR_BGR2GRAY)
    
    # 检测特征点
    orb = cv2.ORB_create(nfeatures=500)
    kp1, des1 = orb.detectAndCompute(ref_gray, None)
    kp2, des2 = orb.detectAndCompute(tgt_gray, None)
    
    # 特征匹配
    bf = cv2.BFMatcher(cv2.NORM_HAMMING, crossCheck=True)
    matches = bf.match(des1, des2)
    matches = sorted(matches, key=lambda x: x.distance)[:50]
    
    # 提取匹配点
    pts1 = np.float32([kp1[m.queryIdx].pt for m in matches])
    pts2 = np.float32([kp2[m.trainIdx].pt for m in matches])
    
    # 估计仿射矩阵（RANSAC）
    M, inliers = cv2.estimateAffinePartial2D(pts2, pts1, method=cv2.RANSAC)
    
    # 变换图像
    h, w = reference.shape[:2]
    aligned = cv2.warpAffine(target, M, (w, h))
    
    return aligned, M
```

#### 光流对齐

处理局部运动，使用稠密光流估计像素级位移：

```python
def align_frames_optical_flow(reference, target):
    """
    基于光流的稠密对齐
    """
    ref_gray = cv2.cvtColor(reference, cv2.COLOR_BGR2GRAY)
    tgt_gray = cv2.cvtColor(target, cv2.COLOR_BGR2GRAY)
    
    # 计算光流 (Farneback)
    flow = cv2.calcOpticalFlowFarneback(
        tgt_gray, ref_gray, None,
        pyr_scale=0.5, levels=3, winsize=15,
        iterations=3, poly_n=5, poly_sigma=1.2, flags=0
    )
    
    # 生成采样网格
    h, w = reference.shape[:2]
    flow_map = np.zeros((h, w, 2), dtype=np.float32)
    flow_map[:,:,0] = np.arange(w)
    flow_map[:,:,1] = np.arange(h)[:, np.newaxis]
    
    # 应用光流位移
    remap_coords = flow_map + flow
    
    # 重采样
    aligned = cv2.remap(target, remap_coords[:,:,0], remap_coords[:,:,1],
                        interpolation=cv2.INTER_LINEAR)
    
    return aligned, flow
```

#### 块匹配对齐

将图像分块，对每块独立搜索最佳匹配：

```python
def align_frames_block_matching(reference, target, block_size=16, search_range=32):
    """
    块匹配对齐
    """
    ref_gray = cv2.cvtColor(reference, cv2.COLOR_BGR2GRAY).astype(np.float32)
    tgt_gray = cv2.cvtColor(target, cv2.COLOR_BGR2GRAY).astype(np.float32)
    
    h, w = ref_gray.shape
    motion_vectors = np.zeros((h // block_size, w // block_size, 2))
    
    for by in range(0, h - block_size + 1, block_size):
        for bx in range(0, w - block_size + 1, block_size):
            # 参考块
            ref_block = ref_gray[by:by+block_size, bx:bx+block_size]
            
            # 在搜索范围内找最佳匹配
            best_sad = float('inf')
            best_mv = (0, 0)
            
            for dy in range(-search_range, search_range + 1):
                for dx in range(-search_range, search_range + 1):
                    ty, tx = by + dy, bx + dx
                    
                    if 0 <= ty < h - block_size and 0 <= tx < w - block_size:
                        tgt_block = tgt_gray[ty:ty+block_size, tx:tx+block_size]
                        sad = np.sum(np.abs(ref_block - tgt_block))
                        
                        if sad < best_sad:
                            best_sad = sad
                            best_mv = (dx, dy)
            
            motion_vectors[by // block_size, bx // block_size] = best_mv
    
    return motion_vectors
```

### 2.3 加权融合策略

运动区域应降低权重以避免拖影：

```python
def weighted_fusion(frames, reference_idx, motion_maps):
    """
    基于运动检测的加权融合
    
    Args:
        frames: 对齐后的帧列表
        reference_idx: 参考帧索引
        motion_maps: 运动强度图列表
    """
    n_frames = len(frames)
    h, w, c = frames[0].shape
    
    # 初始化
    weighted_sum = np.zeros((h, w, c), dtype=np.float64)
    weight_sum = np.zeros((h, w, 1), dtype=np.float64)
    
    for i, frame in enumerate(frames):
        # 计算权重：运动区域权重低
        if i == reference_idx:
            weight = np.ones((h, w, 1), dtype=np.float64)
        else:
            motion = motion_maps[i]
            # 运动强度越大，权重越低
            weight = np.exp(-motion[:,:,np.newaxis] / 50)
        
        weighted_sum += frame.astype(np.float64) * weight
        weight_sum += weight
    
    # 归一化
    result = weighted_sum / (weight_sum + 1e-6)
    
    return np.clip(result, 0, 255).astype(np.uint8)

def compute_motion_map(reference, aligned):
    """计算运动强度图"""
    diff = np.abs(reference.astype(np.float32) - aligned.astype(np.float32))
    motion = np.mean(diff, axis=2)  # 合并RGB通道
    # 平滑
    motion = cv2.GaussianBlur(motion, (15, 15), 0)
    return motion
```

### 2.5 手持长曝光实现

```python
class HandheldLongExposure:
    """手持长曝光模拟"""
    
    def __init__(self, n_frames=8):
        self.n_frames = n_frames
    
    def capture_burst(self, camera):
        """模拟连拍"""
        # 实际应用中从相机获取
        frames = []
        for _ in range(self.n_frames):
            frame = camera.capture()
            frames.append(frame)
        return frames
    
    def process(self, frames):
        """处理连拍序列"""
        reference = frames[len(frames) // 2]  # 中间帧为参考
        
        aligned_frames = [reference]
        motion_maps = [np.zeros(reference.shape[:2])]
        
        # 对齐所有帧
        for i, frame in enumerate(frames):
            if i == len(frames) // 2:
                continue
            
            aligned, flow = align_frames_optical_flow(reference, frame)
            aligned_frames.append(aligned)
            
            # 计算运动图
            motion = np.sqrt(flow[:,:,0]**2 + flow[:,:,1]**2)
            motion_maps.append(motion)
        
        # 加权融合
        result = weighted_fusion(aligned_frames, 0, motion_maps)
        
        # 后处理：锐化
        result = self.sharpen(result)
        
        return result
    
    def sharpen(self, image, amount=0.5):
        """USM锐化"""
        blurred = cv2.GaussianBlur(image, (0, 0), 3)
        sharpened = cv2.addWeighted(image, 1 + amount, blurred, -amount, 0)
        return sharpened
```

## 3. 时域递归降噪（TNR）

### 3.1 IIR时域滤波器模型

时域递归降噪将当前帧与历史结果混合：

$$
O_t = \alpha \cdot I_t + (1 - \alpha) \cdot O_{t-1}
$$

其中：
- $O_t$：输出帧
- $I_t$：输入帧（当前）
- $O_{t-1}$：上一帧输出
- $\alpha$：混合系数（0-1）

```
┌─────────────────────────────────────────────────────────────────┐
│                   TNR 递归滤波结构                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│   输入序列：  I_0, I_1, I_2, I_3, ...                           │
│                 ↓    ↓    ↓    ↓                                │
│                 │    │    │    │                                │
│                 ▼    ▼    ▼    ▼                                │
│   ┌───────┐  ┌─┴─┐┌─┴─┐┌─┴─┐┌─┴─┐                              │
│   │       │  │ α ││ α ││ α ││ α │  ← 混合系数                  │
│   │ Delay ├──┤   ├┤   ├┤   ├┤   │                              │
│   │ (z⁻¹) │  │1-α││1-α││1-α││1-α│                              │
│   └───┬───┘  └─┬─┘└─┬─┘└─┬─┘└─┬─┘                              │
│       │        ↓    ↓    ↓    ↓                                │
│       └────────┴────┴────┴────┘                                │
│                      ↓                                          │
│   输出序列：  O_0, O_1, O_2, O_3, ...                           │
│                                                                  │
│   传递函数： H(z) = α / (1 - (1-α)z⁻¹)                          │
│   等效于一阶低通滤波器                                           │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

### 3.2 运动检测与自适应混合比

静止区域需要强降噪（低α），运动区域需要弱降噪（高α）：

```python
class TemporalRecursiveDenoiser:
    """时域递归降噪"""
    
    def __init__(self, base_alpha=0.2, motion_threshold=20):
        self.base_alpha = base_alpha
        self.motion_threshold = motion_threshold
        self.prev_output = None
        self.prev_input = None
    
    def process_frame(self, current_frame):
        """处理单帧"""
        if self.prev_output is None:
            # 第一帧直接输出
            self.prev_output = current_frame.copy()
            self.prev_input = current_frame.copy()
            return current_frame
        
        # 运动检测
        motion_map = self.detect_motion(current_frame)
        
        # 自适应alpha
        alpha_map = self.compute_adaptive_alpha(motion_map)
        
        # 递归滤波
        output = self.blend(current_frame, self.prev_output, alpha_map)
        
        # 更新状态
        self.prev_output = output.copy()
        self.prev_input = current_frame.copy()
        
        return output
    
    def detect_motion(self, current):
        """基于帧差的运动检测"""
        diff = np.abs(current.astype(np.float32) - 
                      self.prev_input.astype(np.float32))
        
        if len(diff.shape) == 3:
            diff = np.mean(diff, axis=2)
        
        # 平滑运动图
        motion = cv2.GaussianBlur(diff, (11, 11), 0)
        
        return motion
    
    def compute_adaptive_alpha(self, motion_map):
        """
        计算自适应混合系数
        运动大 → alpha高（保留当前帧）
        运动小 → alpha低（时域平均）
        """
        # Sigmoid映射
        normalized = motion_map / self.motion_threshold
        alpha = self.base_alpha + (1 - self.base_alpha) * (
            1 / (1 + np.exp(-5 * (normalized - 1)))
        )
        
        return alpha[:,:,np.newaxis]
    
    def blend(self, current, previous, alpha):
        """加权混合"""
        result = alpha * current.astype(np.float32) + \
                 (1 - alpha) * previous.astype(np.float32)
        return np.clip(result, 0, 255).astype(np.uint8)
```

### 3.3 拖影（Ghosting）问题与对策

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                        拖影问题分析与解决                                    │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│   拖影产生原因：                                                             │
│   ┌────────────────────────────────────────────────────────┐                │
│   │  t-1帧：     ●     （物体在左侧）                      │                │
│   │  t帧：           ●  （物体移到右侧）                   │                │
│   │                                                         │                │
│   │  简单融合后：◐   ●  （左侧残留"鬼影"）                │                │
│   └────────────────────────────────────────────────────────┘                │
│                                                                              │
│   解决策略：                                                                 │
│                                                                              │
│   1. 增强运动检测敏感度                                                      │
│      • 使用更低的阈值                                                        │
│      • 检测边缘运动（不仅是块运动）                                          │
│                                                                              │
│   2. 运动补偿后混合                                                          │
│      • 先将历史帧对齐到当前帧                                                │
│      • 再进行混合                                                            │
│                                                                              │
│   3. 遮挡区域检测                                                            │
│      • 检测新出现/消失的区域                                                 │
│      • 这些区域强制使用当前帧                                                │
│                                                                              │
│   4. 双向运动检测                                                            │
│      • 比较 |I_t - O_{t-1}| 和 |I_t - I_{t-1}|                              │
│      • 选择差异更小的作为混合目标                                            │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

```python
def anti_ghosting_blend(current, prev_output, prev_input, alpha_base=0.3):
    """
    抗拖影融合
    """
    # 当前帧与历史输出的差异
    diff_output = np.abs(current.astype(np.float32) - 
                         prev_output.astype(np.float32))
    
    # 当前帧与历史输入的差异
    diff_input = np.abs(current.astype(np.float32) - 
                        prev_input.astype(np.float32))
    
    if len(diff_output.shape) == 3:
        diff_output = np.mean(diff_output, axis=2)
        diff_input = np.mean(diff_input, axis=2)
    
    # 如果历史输出差异大于历史输入差异，说明可能有拖影
    # 此时应该更信任当前帧
    ghosting_indicator = diff_output > diff_input * 1.5
    
    # 计算自适应alpha
    alpha = np.where(ghosting_indicator, 
                     0.9,  # 拖影区域：几乎完全使用当前帧
                     alpha_base)  # 正常区域：使用基础alpha
    
    alpha = cv2.GaussianBlur(alpha.astype(np.float32), (7, 7), 0)
    alpha = alpha[:,:,np.newaxis]
    
    result = alpha * current.astype(np.float32) + \
             (1 - alpha) * prev_output.astype(np.float32)
    
    return np.clip(result, 0, 255).astype(np.uint8)
```

### 3.4 反馈环路稳定性

TNR是一个反馈系统，需要保证稳定性：

$$
|1 - \alpha| < 1 \Rightarrow 0 < \alpha < 2
$$

实际中 $\alpha \in (0, 1)$ 保证稳定且有意义。

**时间常数分析**：

当 $\alpha = 0.2$ 时：
- 等效时间常数 $\tau = -1/\ln(1-\alpha) \approx 4.5$ 帧
- 意味着约需5帧才能"忘记"历史信息的63%

### 4.2 块匹配（Block Matching）算法

```python
def block_matching_me(current, reference, block_size=16, search_range=16):
    """
    块匹配运动估计
    
    Returns:
        motion_field: (H/BS, W/BS, 2) 运动矢量场
    """
    h, w = current.shape[:2]
    
    if len(current.shape) == 3:
        current = cv2.cvtColor(current, cv2.COLOR_BGR2GRAY)
        reference = cv2.cvtColor(reference, cv2.COLOR_BGR2GRAY)
    
    current = current.astype(np.float32)
    reference = reference.astype(np.float32)
    
    n_blocks_h = h // block_size
    n_blocks_w = w // block_size
    motion_field = np.zeros((n_blocks_h, n_blocks_w, 2), dtype=np.float32)
    
    for by in range(n_blocks_h):
        for bx in range(n_blocks_w):
            y = by * block_size
            x = bx * block_size
            
            # 当前块
            curr_block = current[y:y+block_size, x:x+block_size]
            
            # 全搜索最佳匹配
            best_sad = float('inf')
            best_mv = (0, 0)
            
            for dy in range(-search_range, search_range + 1):
                for dx in range(-search_range, search_range + 1):
                    ry = y + dy
                    rx = x + dx
                    
                    if (0 <= ry < h - block_size and 
                        0 <= rx < w - block_size):
                        ref_block = reference[ry:ry+block_size, 
                                              rx:rx+block_size]
                        sad = np.sum(np.abs(curr_block - ref_block))
                        
                        if sad < best_sad:
                            best_sad = sad
                            best_mv = (dx, dy)
            
            motion_field[by, bx] = best_mv
    
    return motion_field

def motion_compensate(image, motion_field, block_size=16):
    """
    基于运动矢量进行运动补偿
    """
    h, w = image.shape[:2]
    compensated = np.zeros_like(image)
    
    n_blocks_h, n_blocks_w = motion_field.shape[:2]
    
    for by in range(n_blocks_h):
        for bx in range(n_blocks_w):
            y = by * block_size
            x = bx * block_size
            
            dx, dy = motion_field[by, bx]
            
            src_y = int(y + dy)
            src_x = int(x + dx)
            
            if (0 <= src_y < h - block_size and 
                0 <= src_x < w - block_size):
                compensated[y:y+block_size, x:x+block_size] = \
                    image[src_y:src_y+block_size, src_x:src_x+block_size]
            else:
                # 越界使用原位置
                compensated[y:y+block_size, x:x+block_size] = \
                    image[y:y+block_size, x:x+block_size]
    
    return compensated
```

```python
class Denoise3D:
    """3D时空联合降噪"""
    
    def __init__(self, spatial_sigma=2.0, temporal_sigma=4.0, 
                 range_sigma=30, motion_threshold=20):
        self.spatial_sigma = spatial_sigma
        self.temporal_sigma = temporal_sigma
        self.range_sigma = range_sigma
        self.motion_threshold = motion_threshold
        
        self.frame_buffer = []
        self.max_buffer = 5
    
    def process_frame(self, frame):
        """处理单帧"""
        # 添加到缓冲
        self.frame_buffer.append(frame.astype(np.float32))
        if len(self.frame_buffer) > self.max_buffer:
            self.frame_buffer.pop(0)
        
        if len(self.frame_buffer) < 3:
            # 缓冲不足，仅做空域滤波
            return cv2.bilateralFilter(
                frame, d=5, 
                sigmaColor=self.range_sigma,
                sigmaSpace=self.spatial_sigma
            )
        
        # 当前帧索引
        current_idx = len(self.frame_buffer) - 1
        
        # 3D滤波
        result = self.filter_3d(current_idx)
        
        return np.clip(result, 0, 255).astype(np.uint8)
    
    def filter_3d(self, current_idx):
        """3D时空联合滤波"""
        current = self.frame_buffer[current_idx]
        h, w = current.shape[:2]
        
        result = np.zeros_like(current)
        weight_sum = np.zeros((h, w, 1))
        
        for t, frame in enumerate(self.frame_buffer):
            # 时间权重
            dt = abs(t - current_idx)
            temporal_weight = np.exp(-dt**2 / (2 * self.temporal_sigma**2))
            
            # 运动检测
            motion = np.mean(np.abs(frame - current), axis=2)
            motion_mask = motion < self.motion_threshold
            
            # 值域权重
            range_diff = np.mean(np.abs(frame - current), axis=2, keepdims=True)
            range_weight = np.exp(-range_diff**2 / (2 * self.range_sigma**2))
            
            # 组合权重
            weight = temporal_weight * range_weight * motion_mask[:,:,np.newaxis]
            
            result += frame * weight
            weight_sum += weight
        
        # 归一化
        result = result / (weight_sum + 1e-6)
        
        # 空域滤波
        result = cv2.bilateralFilter(
            result.astype(np.float32), d=5,
            sigmaColor=self.range_sigma,
            sigmaSpace=self.spatial_sigma
        )
        
        return result
```
