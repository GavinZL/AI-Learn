### 1.1 高动态范围到低动态范围的压缩问题

### 1.2 保留视觉信息的挑战

色调映射面临的核心矛盾：

| 目标 | 描述 | 挑战 |
|------|------|------|
| **保留对比度** | 场景中不同区域的亮度关系 | 简单压缩会丢失暗部/亮部细节 |
| **保留细节** | 局部纹理和细微变化 | 全局压缩可能消除局部对比 |
| **视觉自然** | 符合人眼适应特性 | 过度处理导致不真实 |
| **避免伪影** | 无光晕、色彩失真 | 局部处理可能引入伪影 |

## 2. 全局色调映射

### 2.1 Reinhard全局算法

Reinhard全局色调映射基于摄影中的**区域曝光系统**：

$$
L_{out}(x,y) = \frac{L_m(x,y)}{1 + L_m(x,y)}
$$

其中 $L_m$ 是经过关键值调整的亮度：

$$
L_m(x,y) = \frac{a}{\bar{L}_w} \cdot L_w(x,y)
$$

#### 关键值估计

关键值 $a$ 控制整体曝光，对数平均亮度用于估计场景关键亮度：

$$
\bar{L}_w = \exp\left(\frac{1}{N}\sum_{x,y}\log(\delta + L_w(x,y))\right)
$$

其中 $\delta$ 是防止对数为负的小常数（如 $10^{-6}$）。

```python
import numpy as np

def reinhard_global(hdr_image, key=0.18, gamma=2.2):
    """
    Reinhard全局色调映射
    
    参数:
        hdr_image: HDR输入图像 (线性光)
        key: 关键值 (0.18为中灰)
        gamma: 输出gamma编码
    返回:
        ldr_image: LDR输出图像
    """
    # 计算亮度
    luminance = 0.2126 * hdr_image[:,:,0] + 0.7152 * hdr_image[:,:,1] + 0.0722 * hdr_image[:,:,2]
    
    # 对数平均亮度
    delta = 1e-6
    log_avg = np.exp(np.mean(np.log(delta + luminance)))
    
    # 缩放到关键值
    Lm = (key / log_avg) * luminance
    
    # Reinhard映射
    Ld = Lm / (1 + Lm)
    
    # 应用到RGB (保持色度)
    scale = Ld / (luminance + delta)
    ldr = hdr_image * scale[:,:,np.newaxis]
    
    # Gamma编码
    ldr = np.clip(ldr, 0, 1)
    ldr = np.power(ldr, 1/gamma)
    
    return (ldr * 255).astype(np.uint8)
```

### 2.2 Filmic曲线

**Filmic色调映射**模拟胶片的响应特性，由John Hable为《Uncharted 2》开发：

$$
f(x) = \frac{(x(Ax+CB)+DE)}{(x(Ax+B)+DF)} - \frac{E}{F}
$$

其中标准参数：
- A = 0.22 (Shoulder Strength)
- B = 0.30 (Linear Strength)
- C = 0.10 (Linear Angle)
- D = 0.20 (Toe Strength)
- E = 0.01 (Toe Numerator)
- F = 0.30 (Toe Denominator)

```

│                    Filmic曲线结构                                │
├─────────────────────────────────────────────────────────────────┤

│       │                          ════                            │
│       │                      ═══                                 │
│   0.8 ┤                   ══                                     │
│       │                ══                                        │
│       │             ══                                           │

│   0.2 ┤ ═                                                        │
│       │═                           Toe                           │
```

```c
// HLSL/GLSL Filmic色调映射
float3 Uncharted2Tonemap(float3 x) {
    float A = 0.22;
    float B = 0.30;
    float C = 0.10;
    float D = 0.20;
    float E = 0.01;
    float F = 0.30;
    
    return ((x * (A * x + C * B) + D * E) / 
            (x * (A * x + B) + D * F)) - E / F;
}

float3 FilmicTonemap(float3 color, float exposure) {
    float3 curr = Uncharted2Tonemap(exposure * color);
    
    // 白点映射
    float W = 11.2;  // Linear White Point
    float3 whiteScale = 1.0 / Uncharted2Tonemap(float3(W, W, W));
    
    return curr * whiteScale;
}
```

```c
    float b = 0.03f;
    float c = 2.43f;
    float d = 0.59f;
    float e = 0.14f;

float3 RRTAndODTFit(float3 v) {
```

### 2.4 S-curve参数化设计

通用的S曲线色调映射：

$$
f(x) = \frac{x^{\gamma_{in}}}{x^{\gamma_{in}} + \sigma^{\gamma_{in}}} \cdot c_{max}
$$

| 参数 | 作用 | 典型值 |
|------|------|--------|
| $\gamma_{in}$ | 曲线陡度 | 1.0-2.0 |
| $\sigma$ | 中点位置 | 0.18-0.5 |
| $c_{max}$ | 最大输出 | 1.0 |

```python
def s_curve_tonemap(x, gamma=1.5, sigma=0.5, c_max=1.0):
    """
    参数化S曲线色调映射
    """
    x_powered = np.power(x, gamma)
    sigma_powered = np.power(sigma, gamma)
    
    return c_max * x_powered / (x_powered + sigma_powered)
```

### 2.5 算法对比

| 算法 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| Reinhard | 简单、自然 | 亮部细节损失 | 一般HDR |
| Filmic | 电影感、层次丰富 | 需要调参 | 游戏、影视 |
| ACES | 工业标准、色彩准确 | 计算稍复杂 | 专业制作 |
| 线性裁剪 | 最简单 | 严重过曝/欠曝 | 不推荐 |

## 3. 局部色调映射

### 3.1 Reinhard局部算法

局部算法使用多尺度环绕函数适应局部亮度：

$$
L_d(x,y) = \frac{L_m(x,y)}{1 + V(x,y,s)}
$$

其中 $V(x,y,s)$ 是选定尺度 $s$ 的局部适应亮度。

```python
import cv2
import numpy as np

def durand_tonemap(hdr_image, sigma_s=2, sigma_r=0.4, base_contrast=5):
    """
    Durand双边色调映射
    
    参数:
        hdr_image: HDR输入 (线性)
        sigma_s: 空间sigma
        sigma_r: 范围sigma
        base_contrast: 目标基础层对比度 (档)
    """
    # 计算亮度
    luminance = 0.2126 * hdr_image[:,:,0] + 0.7152 * hdr_image[:,:,1] + 0.0722 * hdr_image[:,:,2]
    luminance = np.maximum(luminance, 1e-6)
    
    # 对数域
    log_lum = np.log(luminance)
    
    # 双边滤波提取基础层
    # OpenCV的双边滤波需要uint8，这里用自定义或其他库
    log_base = cv2.bilateralFilter(log_lum.astype(np.float32), 
                                    d=0, 
                                    sigmaColor=sigma_r, 
                                    sigmaSpace=sigma_s * max(hdr_image.shape[:2]) / 100)
    
    # 细节层
    log_detail = log_lum - log_base
    
    # 压缩基础层
    log_base_range = log_base.max() - log_base.min()
    target_range = np.log(2) * base_contrast  # 目标档数
    compression = target_range / log_base_range if log_base_range > 0 else 1
    
    log_base_compressed = (log_base - log_base.mean()) * compression + np.log(0.18)
    
    # 合成
    log_output = log_base_compressed + log_detail
    output_lum = np.exp(log_output)
    
    # 恢复颜色
    scale = output_lum / luminance
    output = hdr_image * scale[:,:,np.newaxis]
    
    # 裁剪和gamma
    output = np.clip(output, 0, 1)
    output = np.power(output, 1/2.2)
    
    return (output * 255).astype(np.uint8)
```

### 3.3 Fattal梯度域方法

在对数域对梯度进行衰减：

$$
\nabla H'(x,y) = \phi(x,y) \cdot \nabla H(x,y)
$$

其中衰减因子：

$$
\phi(x,y) = \frac{\alpha}{\|\nabla H(x,y)\|} \left(\frac{\|\nabla H(x,y)\|}{\alpha}\right)^\beta
$$

```c
struct AutoExposure {
    float minEV;      // 最小曝光值 (如 -4)
    float maxEV;      // 最大曝光值 (如 16)
    float adaptSpeed; // 适应速度
    float currentEV;
};

float ComputeAutoExposure(Texture2D luminanceTexture, AutoExposure* ae, float dt) {
    // 从1x1 mipmap读取平均亮度
    float avgLuminance = luminanceTexture.SampleLevel(sampler, float2(0.5, 0.5), 99).r;
    
    // 计算目标EV
    float targetEV = log2(avgLuminance / 0.18);
    targetEV = clamp(targetEV, ae->minEV, ae->maxEV);
    
    // 平滑过渡
    float speed = ae->adaptSpeed * dt;
    ae->currentEV = lerp(ae->currentEV, targetEV, saturate(speed));
    
    return ae->currentEV;
}
```

```python
import cv2

def apply_clahe(image, clip_limit=2.0, tile_size=(8, 8)):
    """
    应用CLAHE进行局部对比度增强
    
    参数:
        image: 输入图像
        clip_limit: 对比度限制
        tile_size: 分块大小
    """
    # 转换到LAB空间
    if len(image.shape) == 3:
        lab = cv2.cvtColor(image, cv2.COLOR_BGR2LAB)
        l, a, b = cv2.split(lab)
    else:
        l = image
    
    # 创建CLAHE对象
    clahe = cv2.createCLAHE(clipLimit=clip_limit, tileGridSize=tile_size)
    
    # 应用到亮度通道
    l_enhanced = clahe.apply(l)
    
    # 合并
    if len(image.shape) == 3:
        enhanced_lab = cv2.merge([l_enhanced, a, b])
        enhanced = cv2.cvtColor(enhanced_lab, cv2.COLOR_LAB2BGR)
    else:
        enhanced = l_enhanced
    
    return enhanced
```

## 5. 色调映射质量评估

### 5.1 TMQI指标

**TMQI**（Tone Mapped Quality Index）综合评估结构保持和自然度：

$$
TMQI = a \cdot S^{\alpha} + (1-a) \cdot N^{\beta}
$$

其中：
- $S$ = 结构保真度（Structure Fidelity）
- $N$ = 统计自然度（Statistical Naturalness）

```python
# TMQI简化实现思路
def compute_tmqi(hdr_image, ldr_image):
    """
    计算TMQI指标
    """
    # 1. 结构保真度 (基于局部标准差比较)
    def local_std(img, window_size=11):
        from scipy.ndimage import uniform_filter
        mean = uniform_filter(img, window_size)
        sqr_mean = uniform_filter(img**2, window_size)
        return np.sqrt(np.maximum(sqr_mean - mean**2, 0))
    
    # HDR和LDR的局部对比度
    hdr_lum = compute_luminance(hdr_image)
    ldr_lum = compute_luminance(ldr_image)
    
    sigma_hdr = local_std(np.log(hdr_lum + 1e-6))
    sigma_ldr = local_std(ldr_lum / 255.0)
    
    # 结构相似性
    C = 0.01
    S = (2 * sigma_hdr * sigma_ldr + C) / (sigma_hdr**2 + sigma_ldr**2 + C)
    S = np.mean(S)
    
    # 2. 统计自然度 (基于亮度直方图)
    hist, _ = np.histogram(ldr_lum.flatten(), bins=256, range=(0, 255))
    hist = hist / hist.sum()
    
    # 与自然图像统计对比（简化：使用高斯拟合）
    mean_target = 115  # 典型自然图像均值
    std_target = 50    # 典型自然图像标准差
    
    ldr_mean = np.mean(ldr_lum)
    ldr_std = np.std(ldr_lum)
    
    N = np.exp(-((ldr_mean - mean_target)**2 / (2 * std_target**2)))
    
    # 3. 综合
    a, alpha, beta = 0.8, 0.3, 0.3
    tmqi = a * (S ** alpha) + (1 - a) * (N ** beta)
    
    return tmqi, S, N
```

### 6.2 OpenGL Compute Shader

```glsl
#version 430

layout(local_size_x = 16, local_size_y = 16) in;

layout(rgba16f, binding = 0) uniform readonly image2D inputImage;
layout(rgba8, binding = 1) uniform writeonly image2D outputImage;

uniform int tonemapOperator; // 0=Reinhard, 1=Filmic, 2=ACES

// Reinhard
vec3 ReinhardTonemap(vec3 color) {
    return color / (1.0 + color);
}

// Uncharted 2 Filmic
vec3 Uncharted2Tonemap(vec3 x) {
    float A = 0.22;
    float B = 0.30;
    float C = 0.10;
    float D = 0.20;
    float E = 0.01;
    float F = 0.30;
    return ((x * (A * x + C * B) + D * E) / (x * (A * x + B) + D * F)) - E / F;
}

vec3 FilmicTonemap(vec3 color) {
    float W = 11.2;
    vec3 curr = Uncharted2Tonemap(color);
    vec3 whiteScale = vec3(1.0) / Uncharted2Tonemap(vec3(W));
    return curr * whiteScale;
}

// ACES
vec3 ACESTonemap(vec3 color) {
    float a = 2.51;
    float b = 0.03;
    float c = 2.43;
    float d = 0.59;
    float e = 0.14;
    return clamp((color * (a * color + b)) / (color * (c * color + d) + e), 0.0, 1.0);
}

void main() {
    ivec2 pixelCoord = ivec2(gl_GlobalInvocationID.xy);
    ivec2 imageSize = imageSize(inputImage);
    
    if (pixelCoord.x >= imageSize.x || pixelCoord.y >= imageSize.y) {
        return;
    }
    
    vec4 hdrColor = imageLoad(inputImage, pixelCoord);
    vec3 exposed = hdrColor.rgb * exposure;
    
    vec3 mapped;
    if (tonemapOperator == 0) {
        mapped = ReinhardTonemap(exposed);
    } else if (tonemapOperator == 1) {
        mapped = FilmicTonemap(exposed);
    } else {
        mapped = ACESTonemap(exposed);
    }
    
    // Gamma
    vec3 output = pow(mapped, vec3(1.0 / 2.2));
    
    imageStore(outputImage, pixelCoord, vec4(output, hdrColor.a));
}
```

```python
    # 1. 估计相机响应曲线
    calibrate = cv2.createCalibrateDebevec()
    response = calibrate.process(exposure_images, exposure_times)
    
    # 2. 合成HDR
    merge = cv2.createMergeDebevec()
    hdr = merge.process(exposure_images, exposure_times, response)
    
    # 3. 色调映射
    tonemap = cv2.createTonemapReinhard(gamma=2.2)
    ldr = tonemap.process(hdr)
    
    return np.clip(ldr * 255, 0, 255).astype(np.uint8)
```
