```java
// Camera2 + ImageReader 图像处理示例
public class ImageProcessor implements ImageReader.OnImageAvailableListener {
    private ImageReader imageReader;
    private HandlerThread processingThread;
    private Handler processingHandler;
    
    public void setupImageReader(int width, int height) {
        // 创建YUV格式的ImageReader
        imageReader = ImageReader.newInstance(
            width, height,
            ImageFormat.YUV_420_888,  // Camera2常用格式
            3  // 最大缓冲图像数
        );
        
        processingThread = new HandlerThread("ImageProcessing");
        processingThread.start();
        processingHandler = new Handler(processingThread.getLooper());
        
        imageReader.setOnImageAvailableListener(this, processingHandler);
    }
    
    @Override
    public void onImageAvailable(ImageReader reader) {
        Image image = reader.acquireLatestImage();
        if (image == null) return;
        
        try {
            // 获取YUV平面
            Image.Plane[] planes = image.getPlanes();
            
            ByteBuffer yBuffer = planes[0].getBuffer();
            ByteBuffer uBuffer = planes[1].getBuffer();
            ByteBuffer vBuffer = planes[2].getBuffer();
            
            int yRowStride = planes[0].getRowStride();
            int uvRowStride = planes[1].getRowStride();
            int uvPixelStride = planes[1].getPixelStride();
            
            // 调用Native处理
            processYUVNative(
                yBuffer, uBuffer, vBuffer,
                image.getWidth(), image.getHeight(),
                yRowStride, uvRowStride, uvPixelStride
            );
            
        } finally {
            image.close();  // 必须关闭以释放缓冲区
        }
    }
    
    private native void processYUVNative(
        ByteBuffer y, ByteBuffer u, ByteBuffer v,
        int width, int height,
        int yRowStride, int uvRowStride, int uvPixelStride
    );
}
```

### 1.3 Android Bitmap/YuvImage处理

```java
// Android Bitmap与YUV处理工具类
public class AndroidImageUtils {
    
    // YUV_420_888 转 NV21
    public static byte[] yuv420ToNv21(Image image) {
        int width = image.getWidth();
        int height = image.getHeight();
        
        Image.Plane[] planes = image.getPlanes();
        ByteBuffer yBuffer = planes[0].getBuffer();
        ByteBuffer uBuffer = planes[1].getBuffer();
        ByteBuffer vBuffer = planes[2].getBuffer();
        
        int ySize = yBuffer.remaining();
        int uSize = uBuffer.remaining();
        int vSize = vBuffer.remaining();
        
        byte[] nv21 = new byte[ySize + uSize + vSize];
        
        // Y平面
        yBuffer.get(nv21, 0, ySize);
        
        // VU交错（NV21格式）
        int uvPixelStride = planes[1].getPixelStride();
        if (uvPixelStride == 2) {
            // 已经是交错格式，直接复制
            vBuffer.get(nv21, ySize, vSize);
        } else {
            // 需要手动交错
            int uvIndex = ySize;
            for (int row = 0; row < height / 2; row++) {
                for (int col = 0; col < width / 2; col++) {
                    int vIndex = row * planes[2].getRowStride() + col;
                    int uIndex = row * planes[1].getRowStride() + col;
                    nv21[uvIndex++] = vBuffer.get(vIndex);
                    nv21[uvIndex++] = uBuffer.get(uIndex);
                }
            }
        }
        
        return nv21;
    }
    
    // NV21 转 Bitmap (使用YuvImage)
    public static Bitmap nv21ToBitmap(byte[] nv21, int width, int height) {
        YuvImage yuvImage = new YuvImage(nv21, ImageFormat.NV21, width, height, null);
        
        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
        yuvImage.compressToJpeg(new Rect(0, 0, width, height), 100, outputStream);
        
        byte[] jpegData = outputStream.toByteArray();
        return BitmapFactory.decodeByteArray(jpegData, 0, jpegData.length);
    }
    
    // Bitmap 处理（使用RenderEffect，Android 12+）
    @RequiresApi(api = Build.VERSION_CODES.S)
    public static void applyBlurEffect(ImageView imageView, float radius) {
        RenderEffect blurEffect = RenderEffect.createBlurEffect(
            radius, radius, Shader.TileMode.CLAMP
        );
        imageView.setRenderEffect(blurEffect);
    }
}
```

### 1.4 NDK中的图像处理

```c
// JNI + libyuv 图像处理
#include <jni.h>
#include <libyuv.h>
#include <arm_neon.h>

extern "C" JNIEXPORT void JNICALL
Java_com_example_ImageProcessor_processYUVNative(
    JNIEnv *env, jobject thiz,
    jobject y_buffer, jobject u_buffer, jobject v_buffer,
    jint width, jint height,
    jint y_stride, jint uv_stride, jint uv_pixel_stride)
{
    uint8_t* y_data = (uint8_t*)env->GetDirectBufferAddress(y_buffer);
    uint8_t* u_data = (uint8_t*)env->GetDirectBufferAddress(u_buffer);
    uint8_t* v_data = (uint8_t*)env->GetDirectBufferAddress(v_buffer);
    
    // 分配输出缓冲区
    int dst_size = width * height * 4;  // ARGB
    uint8_t* argb_data = (uint8_t*)malloc(dst_size);
    
    if (uv_pixel_stride == 1) {
        // I420格式
        libyuv::I420ToARGB(
            y_data, y_stride,
            u_data, uv_stride,
            v_data, uv_stride,
            argb_data, width * 4,
            width, height
        );
    } else {
        // NV12/NV21格式
        libyuv::NV21ToARGB(
            y_data, y_stride,
            v_data, uv_stride,  // V在前
            argb_data, width * 4,
            width, height
        );
    }
    
    // 应用NEON优化的滤波
    apply_gaussian_blur_neon(argb_data, width, height);
    
    free(argb_data);
}

// NEON优化的灰度转换
void rgb_to_gray_neon(const uint8_t* rgba, uint8_t* gray, int count) {
    // R*0.299 + G*0.587 + B*0.114
    // 使用定点: R*77 + G*150 + B*29 >> 8
    
    uint8x8_t coef_r = vdup_n_u8(77);
    uint8x8_t coef_g = vdup_n_u8(150);
    uint8x8_t coef_b = vdup_n_u8(29);
    
    for (int i = 0; i < count - 7; i += 8) {
        // 加载8个RGBA像素
        uint8x8x4_t rgba_vec = vld4_u8(rgba + i * 4);
        
        // 计算灰度
        uint16x8_t sum = vmull_u8(rgba_vec.val[0], coef_r);  // R
        sum = vmlal_u8(sum, rgba_vec.val[1], coef_g);         // G
        sum = vmlal_u8(sum, rgba_vec.val[2], coef_b);         // B
        
        // 右移8位
        uint8x8_t result = vshrn_n_u16(sum, 8);
        
        vst1_u8(gray + i, result);
    }
}
```

```swift
class CoreImageProcessor {
    private let context: CIContext
    
    init() {
        // 创建Metal加速的Context
        if let device = MTLCreateSystemDefaultDevice() {
            context = CIContext(mtlDevice: device, options: [
                .cacheIntermediates: false,  // 减少内存使用
                .priorityRequestLow: false   // 高优先级处理
            ])
        } else {
            context = CIContext()
        }
    }
    
    // 使用内置滤镜链
    func applyFilters(to image: CIImage) -> CIImage {
        // 1. 高斯模糊
        let blurFilter = CIFilter.gaussianBlur()
        blurFilter.inputImage = image
        blurFilter.radius = 5.0
        
        // 2. 色彩调整
        let colorFilter = CIFilter.colorControls()
        colorFilter.inputImage = blurFilter.outputImage
        colorFilter.saturation = 1.2
        colorFilter.brightness = 0.05
        colorFilter.contrast = 1.1
        
        // 3. 锐化
        let sharpenFilter = CIFilter.sharpenLuminance()
        sharpenFilter.inputImage = colorFilter.outputImage
        sharpenFilter.sharpness = 0.4
        
        return sharpenFilter.outputImage ?? image
    }
    
    // 渲染到CGImage
    func render(_ ciImage: CIImage) -> CGImage? {
        let extent = ciImage.extent
        return context.createCGImage(ciImage, from: extent)
    }
    
    // 零拷贝渲染到CVPixelBuffer
    func render(_ ciImage: CIImage, to pixelBuffer: CVPixelBuffer) {
        context.render(ciImage, to: pixelBuffer)
    }
```

### 2.2 CIKernel自定义滤镜

```swift
// 自定义CIKernel (Metal Shading Language)
class CustomFilter: CIFilter {
    @objc dynamic var inputImage: CIImage?
    @objc dynamic var inputRadius: Float = 5.0
    
    // Metal Kernel代码
    static let kernelSource = """
    #include <CoreImage/CoreImage.h>
    
    extern "C" float4 customBlur(coreimage::sampler src, float radius) {
        float2 dc = src.coord();
        float4 sum = float4(0.0);
        float weight = 0.0;
        
        for (float y = -radius; y <= radius; y += 1.0) {
            for (float x = -radius; x <= radius; x += 1.0) {
                float dist = length(float2(x, y));
                if (dist <= radius) {
                    float w = exp(-dist * dist / (2.0 * radius * radius));
                    sum += src.sample(dc + float2(x, y)) * w;
                    weight += w;
                }
            }
        }
        
        return sum / weight;
    }
    """
    
    private static var kernel: CIKernel? = {
        try? CIKernel(functionName: "customBlur", 
                      fromMetalLibraryData: compileMetalSource(kernelSource))
    }()
    
    override var outputImage: CIImage? {
        guard let input = inputImage,
              let kernel = Self.kernel else { return nil }
        
        return kernel.apply(
            extent: input.extent,
            roiCallback: { _, rect in
                rect.insetBy(dx: CGFloat(-inputRadius), dy: CGFloat(-inputRadius))
            },
            arguments: [input, inputRadius]
        )
    }
}
```

```swift
import Accelerate

class vImageProcessor {
    // 高斯模糊
    func gaussianBlur(image: CGImage, radius: UInt32) -> CGImage? {
        guard var sourceBuffer = try? vImage_Buffer(cgImage: image) else { return nil }
        
        var destBuffer = try? vImage_Buffer(
            width: Int(sourceBuffer.width),
            height: Int(sourceBuffer.height),
            bitsPerPixel: 32
        )
        
        guard var dest = destBuffer else { return nil }
        
        // 计算kernel大小（必须为奇数）
        var kernelSize = radius | 1
        
        // 应用卷积
        let error = vImageConvolve_ARGB8888(
            &sourceBuffer,
            &dest,
            nil,
            0, 0,
            createGaussianKernel(radius: Int(kernelSize)),
            kernelSize,
            kernelSize,
            Int32(kernelSize * kernelSize),
            nil,
            vImage_Flags(kvImageEdgeExtend)
        )
        
        guard error == kvImageNoError else { return nil }
        
        defer {
            sourceBuffer.free()
            dest.free()
        }
        
        return try? dest.createCGImage(format: vImage_CGImageFormat(cgImage: image)!)
    }
    
    // 色彩空间转换
    func convertRGBToYUV(rgbImage: CGImage) -> (y: vImage_Buffer, 
                                                  u: vImage_Buffer, 
                                                  v: vImage_Buffer)? {
        guard var source = try? vImage_Buffer(cgImage: rgbImage) else { return nil }
        
        let width = Int(source.width)
        let height = Int(source.height)
        
        var yBuffer = try? vImage_Buffer(width: width, height: height, bitsPerPixel: 8)
        var uBuffer = try? vImage_Buffer(width: width/2, height: height/2, bitsPerPixel: 8)
        var vBuffer = try? vImage_Buffer(width: width/2, height: height/2, bitsPerPixel: 8)
        
        guard var y = yBuffer, var u = uBuffer, var v = vBuffer else { return nil }
        
        // RGB to YUV转换矩阵 (BT.601)
        var matrix = vImage_YpCbCrPixelRange(
            Yp_bias: 16,
            CbCr_bias: 128,
            YpRangeMax: 235,
            CbCrRangeMax: 240,
            YpMax: 255,
            YpMin: 0,
            CbCrMax: 255,
            CbCrMin: 0
        )
        
        var info = vImage_ARGBToYpCbCr()
        vImageConvert_ARGBToYpCbCr_GenerateConversion(
            kvImage_ARGBToYpCbCrMatrix_ITU_R_601_4,
            &matrix,
            &info,
            kvImageARGB8888,
            kvImage420Yp8_Cb8_Cr8,
            vImage_Flags(kvImageNoFlags)
        )
        
        var destBuffers = [y, u, v]
        vImageConvert_ARGB8888ToYpCbCr(
            &source,
            &destBuffers[0],
            &destBuffers[1],
            &destBuffers[2],
            &info,
            nil,
            vImage_Flags(kvImageNoFlags)
        )
        
        return (destBuffers[0], destBuffers[1], destBuffers[2])
    }
    
    // 直方图均衡化
    func histogramEqualization(image: CGImage) -> CGImage? {
        guard var source = try? vImage_Buffer(cgImage: image) else { return nil }
        var dest = try? vImage_Buffer(
            width: Int(source.width),
            height: Int(source.height),
            bitsPerPixel: 32
        )
        guard var destination = dest else { return nil }
        
        vImageEqualization_ARGB8888(&source, &destination, vImage_Flags(kvImageNoFlags))
        
        defer {
```

```cpp
            dst_u, dst_stride_u,
            dst_v, dst_stride_v,
            width, height
        );
    }
    
    // I420 转 ARGB
    static int I420ToARGB(const uint8_t* src_y, int src_stride_y,
                          const uint8_t* src_u, int src_stride_u,
                          const uint8_t* src_v, int src_stride_v,

                          int width, int height) {
        return libyuv::I420ToARGB(
            src_y, src_stride_y,
            src_u, src_stride_u,
            src_v, src_stride_v,
            dst_argb, dst_stride_argb,
            width, height
        );
    }
    
    // 缩放
    static int Scale(const uint8_t* src_y, int src_stride_y,
                     const uint8_t* src_u, int src_stride_u,
                     const uint8_t* src_v, int src_stride_v,
                     int src_width, int src_height,
                     uint8_t* dst_y, int dst_stride_y,
                     uint8_t* dst_u, int dst_stride_u,
                     uint8_t* dst_v, int dst_stride_v,

                     libyuv::FilterMode mode) {
        return libyuv::I420Scale(
            src_y, src_stride_y,
            src_u, src_stride_u,
            src_v, src_stride_v,
            src_width, src_height,
            dst_y, dst_stride_y,
            dst_u, dst_stride_u,
            dst_v, dst_stride_v,
            dst_width, dst_height,
            mode  // kFilterNone, kFilterLinear, kFilterBilinear, kFilterBox
        );
    }
    
    // 旋转
    static int Rotate(const uint8_t* src_y, int src_stride_y,
                      const uint8_t* src_u, int src_stride_u,
                      const uint8_t* src_v, int src_stride_v,

            dst_y, dst_stride_y,

            dst_v, dst_stride_v,
            width, height,
            rotation  // kRotate0, kRotate90, kRotate180, kRotate270

    }
};
```

### 4.3 条件编译与运行时检测

```cpp
// CPU特性检测
#include <cpuinfo.h>  // 或自定义检测

class CPUFeatures {
public:
    static CPUFeatures& instance() {
        static CPUFeatures inst;
        return inst;
    }
    
    bool hasNEON() const { return neon_supported; }
    bool hasSSE2() const { return sse2_supported; }
    bool hasSSE4() const { return sse4_supported; }
    bool hasAVX2() const { return avx2_supported; }
    bool hasAVX512() const { return avx512_supported; }
    
private:
    CPUFeatures() {
        detect();
    }
    
    void detect() {
#if defined(__ARM_NEON) || defined(__ARM_NEON__)
        neon_supported = true;
#elif defined(__x86_64__) || defined(_M_X64)
        // 使用CPUID检测
        detectx86Features();
#endif
    }
    
    void detectx86Features() {
#ifdef _MSC_VER
        int cpuInfo[4];
        __cpuid(cpuInfo, 1);
        sse2_supported = (cpuInfo[3] & (1 << 26)) != 0;
        sse4_supported = (cpuInfo[2] & (1 << 19)) != 0;
        
        __cpuid(cpuInfo, 7);
        avx2_supported = (cpuInfo[1] & (1 << 5)) != 0;
        avx512_supported = (cpuInfo[1] & (1 << 16)) != 0;
#else
        __builtin_cpu_init();
        sse2_supported = __builtin_cpu_supports("sse2");
        sse4_supported = __builtin_cpu_supports("sse4.1");
        avx2_supported = __builtin_cpu_supports("avx2");
        avx512_supported = __builtin_cpu_supports("avx512f");
#endif
    }
    
    bool neon_supported = false;
    bool sse2_supported = false;
    bool sse4_supported = false;
    bool avx2_supported = false;
    bool avx512_supported = false;
};

// 运行时分派
void gaussianBlur(const uint8_t* src, uint8_t* dst, 
                  int width, int height, float sigma) {
    auto& cpu = CPUFeatures::instance();
    
#if defined(__ARM_NEON)
    if (cpu.hasNEON()) {
        gaussianBlur_NEON(src, dst, width, height, sigma);
        return;
    }
#elif defined(__x86_64__)
    if (cpu.hasAVX512()) {
        gaussianBlur_AVX512(src, dst, width, height, sigma);
        return;
    }
    if (cpu.hasAVX2()) {
        gaussianBlur_AVX2(src, dst, width, height, sigma);
        return;
    }
    if (cpu.hasSSE4()) {
        gaussianBlur_SSE4(src, dst, width, height, sigma);
        return;
    }
#endif
    
    // 回退到标量实现
    gaussianBlur_Scalar(src, dst, width, height, sigma);
}
```
